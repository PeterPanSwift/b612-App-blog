class NumberFormatter2: NumberFormatter {
    override func getObjectValue(_ obj: AutoreleasingUnsafeMutablePointer<AnyObject?>?, for string: String, range rangep: UnsafeMutablePointer<NSRange>?) throws {
        if let number = Double(string) {
            obj?.pointee = number as AnyObject
        } else {
            obj?.pointee = 0 as AnyObject
        }
    }
}

struct ContentView: View {
    @State private var salary: Double = 500.0 //
    var body: some View {
        VStack {
            Text("\(salary, specifier: "%.2f")")
            TextField("0", value: $salary, formatter: NumberFormatter2())
                .keyboardType(.decimalPad)
                .foregroundStyle(Color.mint)
                .frame(width: 200)
                .font(.system(size: 26))
                .textFieldStyle(.roundedBorder)
                .padding()
        }
    }
}
