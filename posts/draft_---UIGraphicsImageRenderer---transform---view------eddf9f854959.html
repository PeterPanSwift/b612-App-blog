利用 UIGraphicsImageRenderer 將被transform 的 view 變成圖片


Draw UIView with transform into UIImage

class ViewController: UIViewController {
    
    @IBOutlet weak var imageViewContainer: UIView!
    @IBOutlet weak var imageView1: UIImageView!
    @IBOutlet weak var imageView2: UIImageView!
    override func viewDidLoad() {
        super.viewDidLoad()
        // Do any additional setup after loading the view.
        
        imageView1.transform = CGAffineTransform(rotationAngle: .pi / 180 * 180)
        
    }
    
    
    @IBAction func tap(_ sender: Any) {
        
        let renderer = UIGraphicsImageRenderer(size: imageViewContainer.bounds.size)
        let image = renderer.image(actions: { (context) in
            imageViewContainer.drawHierarchy(in: imageViewContainer.bounds, afterScreenUpdates: true)
        })
        imageView2.image = image
    }
}
