使用 SwiftUI Toggle 實現選取狀態的 button


struct ContentView: View {
    @State private var italic = false

    var body: some View {
        Toggle(isOn: $italic) {
            Text("I")
        }
        .toggleStyle(.button)
    }
}


struct ContentView: View {
    @State private var italic = false

    var body: some View {
        Toggle(isOn: $italic) {
            Text("I")
        }
        .toggleStyle(.button)
        .buttonStyle(.bordered)
    }
}
