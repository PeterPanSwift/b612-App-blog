struct Game: Identifiable {
    var id = UUID()
    var date: String
    var guestTeam: String
    var homeTeam: String
    var time: String
    var venue: String

    var body: some View {
        VStack(alignment: .leading) {
            Text("\\(date) \\(time)")
                .font(.subheadline)
            Text("\\(guestTeam) vs \\(homeTeam)")
                .font(.title2)
            Text(venue)
                .font(.subheadline)
        }
    }
}
