import SwiftUI

struct AgeCalculatorView: View {
    @State private var birthday = Date()
    @State private var age: Int? = nil
    @State private var showDatePicker = false

    var body: some View {
        ZStack { // ç”¨ ZStack ç–Šèµ·ä¾†ï¼Œæ‰èƒ½è®“ DatePicker é£„æµ®ï½
            VStack(spacing: 30) {
                Text("HOW OLD ARE YOU?")
                    .font(.largeTitle)
                    .bold()
                    .padding(.top, 40)

                Button(action: {
                    withAnimation {
                        showDatePicker = true
                    }
                }) {
                    HStack {
                        Text("SELECT YOUR BIRTHDAY")
                        Image(systemName: "birthday.cake.fill")
                    }
                    .padding()
                    .frame(maxWidth: .infinity)
                }
                .buttonStyle(.borderedProminent)
                .padding(.horizontal, 30)

                if let age = age {
                    VStack(spacing: 10) {
                        Text("YOU ARE")
                            .font(.title2)
                            .bold()
                        Text("\(age) YEAR\(age == 1 ? "" : "S") OLD!")
                            .font(.largeTitle)
                            .bold()
                            .foregroundColor(.orange)

                        Image("fox_with_party_hat")
                            .resizable()
                            .scaledToFit()
                            .frame(width: 150, height: 150)
                            .transition(.scale)
                    }
                    .padding(.top, 20)
                }

                Spacer()
            }
            .padding()

            // --- æµ®èµ·ä¾†çš„ DatePicker ---
            if showDatePicker {
                Color.black.opacity(0.4) // èƒŒæ™¯é®ç½©ï¼Œåƒæ˜¯æ™šä¸Šæ˜Ÿæ˜Ÿé–ƒé–ƒğŸŒŸ
                    .ignoresSafeArea()
                    .onTapGesture {
                        withAnimation {
                            showDatePicker = false
                        }
                    }

                VStack(spacing: 20) {
                    DatePicker(
                        "Select your birthday",
                        selection: $birthday,
                        displayedComponents: .date
                    )
                    .datePickerStyle(.wheel)
                    .labelsHidden()
                    .padding()

                    Button("CALCULATE AGE ğŸˆ") {
                        calculateAge()
                        withAnimation {
                            showDatePicker = false
                        }
                    }
                    .buttonStyle(.borderedProminent)
                }
                .padding()
                .background(
                    RoundedRectangle(cornerRadius: 20)
                        .fill(Color.white)
                        .shadow(radius: 10)
                )
                .padding(40)
                .transition(.scale)
            }
        }
    }

    private func calculateAge() {
        let calendar = Calendar.current
        let now = Date()
        let components = calendar.dateComponents([.year], from: birthday, to: now)
        age = components.year
    }
}
