使用 Swift Package Manager 安裝套件


  1. swift package init — type executable

2. 編編 Package.swift，加入套件 SwiftyJSON

// swift-tools-version:3.1
import PackageDescription
let package = Package(
  name: "Demo",
  dependencies: [
     .Package(url: “https://github.com/SwiftyJSON/SwiftyJSON.git",   
         majorVersion: 3, minor: 1)
  ]
)

3. 下載套件

swift package fetch

4. swift build

5. swift package generate-xcodeproj
