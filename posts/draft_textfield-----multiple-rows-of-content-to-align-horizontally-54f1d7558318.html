textfield 的對齊 multiple rows of content to align horizontally


struct ContentView: View {
    @State private var name = ""
    @State private var bestFriend = ""

    var body: some View {
        Form {
            HStack {
                Text("name")
                TextField("name", text: $name)
                    .textFieldStyle(.roundedBorder)
            }
            HStack {
                Text("best friend")
                TextField("bestFriend", text: $name)
                    .textFieldStyle(.roundedBorder)
            }
        }
    }
}
struct ContentView: View {
    @State private var name = ""
    @State private var bestFriend = ""

    var body: some View {
        Form {
            Grid(alignment: .leading) {
                GridRow {
                    Text("name")
                    TextField("name", text: $name)
                        .textFieldStyle(.roundedBorder)
                }
                GridRow {
                    Text("best friend")
                    TextField("bestFriend", text: $name)
                        .textFieldStyle(.roundedBorder)
                }
            }
            
        }
    }
}
