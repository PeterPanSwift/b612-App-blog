SwiftUI 多個 sheet 問題

d

VStack {
            Button(action: {
                self.showA = true
            }) {
                Text("A")
            }
            Button(action: {
                self.showB = true
            }) {
                Text("B")
            }
            
        }
        .sheet(isPresented: $showA) {
            Text("A")
        }
        .sheet(isPresented: $showB) {
            Text("B")
        }

解答

VStack {
            Button(action: {
                self.showA = true
            }) {
                Text("A")
            }
            .sheet(isPresented: $showA) {
                Text("A")
            }
            Button(action: {
                self.showB = true
            }) {
                Text("B")
            }
            .sheet(isPresented: $showB) {
                Text("B")
            }
            
        }
