#229 使用 property 客製 SwiftUI view

Apple 的 Customize views with properties 設計了一個漂亮的每日天氣，介紹如何用 property 客製可重覆利用的 SwiftUI view。

import SwiftUI

struct ContentView: View {
    var body: some View {
        HStack {
            DayForecast(day: "Mon", isRainy: false, high: 70, low: 50)
            DayForecast(day: "Tue", isRainy: true, high: 60, low: 40)
        }
    }
}

struct DayForecast: View {
    let day: String
    let isRainy: Bool
    let high: Int
    let low: Int
    
    var iconName: String {
        if isRainy {
            return "cloud.rain.fill"
        } else {
            return "sun.max.fill"
        }
    }
    
    var iconColor: Color {
        if isRainy {
            return Color.blue
        } else {
            return Color.yellow
        }
    }
    
    var body: some View {
        VStack {
            Text(day)
                .font(Font.headline)
            Image(systemName: iconName)
                .foregroundStyle(iconColor)
                .font(Font.largeTitle)
            Text("High: \(high)")
                .fontWeight(Font.Weight.semibold)
            Text("Low: \(low)")
                .fontWeight(Font.Weight.medium)
        }
        .padding()
    }
}

struct CharacterIntroduction: View {
    let name: String
    let imageName: String
    let description: String
    let isGood: Bool
    
    var morality: String {
        isGood ? "好人" : "壞人"
    }
    
    var moralityColor: Color {
        isGood ? .blue : .red
    }
    
    var body: some View {
        VStack {
            Text(name)
                .font(Font.headline)
            Text(morality)
                .foregroundStyle(moralityColor)
            Image(imageName)
                .resizable()
                .scaledToFill()
                .frame(width: 100, height: 100)
                .clipShape(.circle)
            Text(description)
                .font(.subheadline)
                .padding()
        }
        .padding()
    }
}

import SwiftUI

struct ContentView: View {
    var body: some View {
        HStack {
            MatchScore(team1: "Team A", team2: "Team B", score1: 3, score2: 2)
            MatchScore(team1: "Team C", team2: "Team D", score1: 1, score2: 1)
        }
    }
}

struct MatchScore: View {
    let team1: String
    let team2: String
    let score1: Int
    let score2: Int
    
    var body: some View {
        VStack {
            Text("\(team1) vs \(team2)")
                .font(.headline)
            HStack {
                Text("\(score1)")
                    .font(.largeTitle)
                    .fontWeight(.bold)
                Text("-")
                    .font(.largeTitle)
                    .fontWeight(.bold)
                Text("\(score2)")
                    .font(.largeTitle)
                    .fontWeight(.bold)
            }
            .padding()
            Text(matchResult)
                .font(.subheadline)
                .foregroundStyle(resultColor)
        }
        .padding()
    }
    
    var matchResult: String {
        if score1 > score2 {
            "\(team1) Wins"
        } else if score1 < score2 {
            "\(team2) Wins"
        } else {
            "Draw"
        }
    }
    
    var resultColor: Color {
        if score1 > score2 {
            .green
        } else if score1 < score2 {
            .red
        } else {
            .gray
        }
    }
}
import SwiftUI

struct ContentView: View {
    var body: some View {
        HStack {
            BookRecommendation(title: "The Little Prince", author: "Antoine de Saint-Exupéry", isFavorite: true)
            BookRecommendation(title: "1984", author: "George Orwell", isFavorite: false)
        }
    }
}

struct BookRecommendation: View {
    let title: String
    let author: String
    let isFavorite: Bool
    
    var body: some View {
        VStack {
            Text(title)
                .font(.headline)
            Text("by \(author)")
                .font(.subheadline)
            Image(systemName: iconName)
                .foregroundStyle(iconColor)
                .font(.largeTitle)
        }
        .padding()
    }

    var iconName: String {
        if isFavorite {
            "heart.fill"
        } else {
            "heart"
        }
    }
    
    var iconColor: Color {
        if isFavorite {
            .red
        } else {
            .gray
        }
    }
}
