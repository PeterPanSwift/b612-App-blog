顯示輸入的文字字數和限制字數


TextField("Your words here...",text: $words)
                    .frame(width: 170, height: 0, alignment: .center)
                    .padding()
                    .border(Color.black, width: 1)
                    .fontWeight(.bold)
                    .onChange(of: words) { newValue in
                        if newValue.count > 6 {
                            words = String(words.prefix(6))
                        }
                    }
