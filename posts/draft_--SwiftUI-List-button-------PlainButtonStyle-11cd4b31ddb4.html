讓 SwiftUI List button 可以點選的 PlainButtonStyle


struct SongList: View {
    let songs = Song.data
    var body: some View {
        List(songs, id: \.name) { (song) in
            SongRow(song: song)
        }
    }
}

struct SongRow: View {
    let song: Song
    
    var body: some View {
        HStack(spacing: 10) {
            Image(song.albumImage)
                .resizable()
                .scaledToFill()
                .frame(width: 40, height: 40)
                .cornerRadius(6)
            VStack(alignment: .leading, spacing: 4) {
                Text(song.name)
                Text(song.singer)
                    .font(.caption)
                    .foregroundColor(.secondary)
                    
            }
            Spacer()
            Button(action: {
                print("tap")
            }, label: {
                Image(systemName: "plus")
                    .foregroundColor(.red)
                
            })
            .buttonStyle(PlainButtonStyle())
        }
    }
}
