How do you decode a JSON feed that has several custom date formats? If you are lucky using a dateDecodingStrategy might…useyourloaf.com


struct Product: Codable {
var created_at: Date

}


let task = URLSession.shared.dataTask(with: URL(string: "http://makeup-api.herokuapp.com/api/v1/products.json")!) { (data, response, error) in
if let data = data {
let decoder = JSONDecoder()
decoder.dateDecodingStrategy = .iso8601
do {
let products = try decoder.decode([Product].self, from: data)
print(products.first)

} catch {
print(error)
}

}
}
task.resume()
