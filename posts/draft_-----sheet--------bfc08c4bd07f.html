利用一個 sheet 顯示不同頁面


enum Animal: Identifiable {
    case cat
    case dog
    var id: Self { self }
}
struct ContentView: View {
    @State private var animal: Animal?
    
    var body: some View {
        VStack {
            Button(action: {
                animal = .cat
            }, label: {
                Text("Cat")
            })
            
            Button(action: {
                animal = .dog
            }, label: {
                Text("Dog")
            })
        }
        .sheet(item: $animal) { item in
            switch item {
            case .cat:
                CatView()
            case .dog:
                DogView()
            }
        }
        
    }
}
