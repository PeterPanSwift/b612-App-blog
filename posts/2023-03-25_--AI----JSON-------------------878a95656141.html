import Foundation

struct FeedWrapper: Codable {
    let feed: Feed
}

struct Feed: Codable {
    let title: String
    let id: URL
    let author: Author
    let links: [Link]
    let copyright: String
    let country: String
    let icon: URL
    let updated: Date
    let results: [Result]
}

struct Author: Codable {
    let name: String
    let url: URL
}

struct Link: Codable {
    let `self`: URL
}

struct Result: Codable {
    let artistName: String
    let id: String
    let name: String
    let releaseDate: Date
    let kind: String
    let artistId: String
    let artistUrl: URL
    let contentAdvisoryRating: String?
    let artworkUrl100: URL
    let genres: [Genre]
    let url: URL
}

struct Genre: Codable {
    let genreId: String
    let name: String
    let url: URL
}

// Date Formatter
let dateFormatter: DateFormatter = {
    let formatter = DateFormatter()
    formatter.dateFormat = "EEE, dd MMM yyyy HH:mm:ss Z"
    return formatter
}()

// JSON Decoder
let decoder = JSONDecoder()
decoder.dateDecodingStrategy = .formatted(dateFormatter)

do {
    let feedWrapper = try decoder.decode(FeedWrapper.self, from: jsonData)
    print(feedWrapper)
} catch {
    print("Error decoding JSON: \(error)")
}

根據您提供的 JSON，您可以使用以下 Swift Codable 型別來定義您需要的資料：

import Foundation

struct AlbumResponse: Codable {
    let feed: Feed
}

struct Feed: Codable {
    let results: [Album]
}

struct Album: Codable {
    let artistName: String
    let name: String
    let releaseDate: Date
    let artworkUrl100: URL
}

// 使用 DateFormatter 來解析日期
let dateFormatter = DateFormatter()
dateFormatter.dateFormat = "yyyy-MM-dd"

// 使用 JSONDecoder 來解析 JSON
let jsonDecoder = JSONDecoder()
jsonDecoder.dateDecodingStrategy = .formatted(dateFormatter)

do {
    let albumResponse = try jsonDecoder.decode(AlbumResponse.self, from: jsonData)
    print(albumResponse)
} catch {
    print("Error decoding JSON: \(error)")
}

這裡我們創建了 AlbumResponse、Feed 和 Album 結構，並且只保留了您所需的屬性。另外，我們使用了 DateFormatter 和 JSONDecoder 來解析日期和網址。

使用 Claude，上傳 JSON 檔。

class Song {
  final String artistName;
  final String trackName;
  final DateTime releaseDate;
  final Uri artworkUrl100;

  Song({
    required this.artistName,
    required this.trackName,
    required this.releaseDate,
    required this.artworkUrl100,
  });

  factory Song.fromJson(Map<String, dynamic> json) {
    return Song(
      artistName: json['artistName'],
      trackName: json['trackName'],
      releaseDate: DateTime.parse(json['releaseDate']),
      artworkUrl100: Uri.parse(json['artworkUrl100']),
    );
  }
}
