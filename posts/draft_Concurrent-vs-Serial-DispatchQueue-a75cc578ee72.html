不會閃退

let serialQueue = DispatchQueue(label: "test1")
let concurrentQueue = DispatchQueue(label: "test2", attributes: .concurrent)
serialQueue.async {
    print("async1")
    Thread.sleep(forTimeInterval: 1)
    print("async2")
}
print("aaa")
serialQueue.sync {
    print("sync")
}

閃退

serialQueue.async {
    print("async1")
    serialQueue.sync {
        print("sync")
    }
    print("async2")
}
