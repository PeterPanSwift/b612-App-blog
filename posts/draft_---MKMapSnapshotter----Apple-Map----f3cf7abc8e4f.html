使用 MKMapSnapshotter 產生 Apple Map 圖片

使用 MKMapSnapshotter 產生 Apple Map 圖片


import MapKit
import UIKit

class ViewController: UIViewController {

    @IBOutlet weak var mapImageView: UIImageView!
    
    override func viewDidLoad() {
        super.viewDidLoad()
        // Do any additional setup after loading the view.
        
        generateMapSnapshot()
    }

    func generateMapSnapshot() {
        // 設置地圖的中心點和範圍
        let coordinate = CLLocationCoordinate2D(
            latitude: 25.0330,
            longitude: 121.5654
        )
        let region = MKCoordinateRegion(
            center: coordinate,
            latitudinalMeters: 1000,
            longitudinalMeters: 1000
        )

        // 配置快照選項
        let options = MKMapSnapshotter.Options()
        options.region = region
        options.size = CGSize(width: 400, height: 400)  // 圖片尺寸
        options.scale = UIScreen.main.scale  // 適配 Retina 顯示
        options.mapType = .standard  // 地圖類型（可設為 .satellite 或 .hybrid）

        // 創建快照器並生成圖片
        let snapshotter = MKMapSnapshotter(options: options)
        snapshotter.start { snapshot, error in
            guard let snapshot,
                error == nil
            else {
                print(
                    "Error generating snapshot: \(error?.localizedDescription ?? "Unknown error")"
                )
                return
            }

            // 獲取生成的圖片
            let image = snapshot.image
            self.mapImageView.image = image
        }
    }

}
