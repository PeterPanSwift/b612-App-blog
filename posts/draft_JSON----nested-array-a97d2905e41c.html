uHunt API
uHunt API is a web service that provides real time update on submissions and statistics of UVa users.uhunt.onlinejudge.org


struct Item {
    let number: Int
    let letter: String
}

extension Item: Codable {
    init(from decoder: Decoder) throws {
        var container = try decoder.unkeyedContainer()
        let _ = try container.decode(Int.self) // 跳過第一個元素
        let letter = try container.decode(String.self)
        let number = try container.decode(Int.self)
        self.init(number: number, letter: letter)
    }
}

let data = """
[
    [1, "a", 3],
    [2, "b", 4],
]
""".data(using: .utf8)!

let decoder = JSONDecoder()
do {
