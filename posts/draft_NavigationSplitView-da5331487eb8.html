NavigationSplitView


struct Song: Hashable {
    let name: String
}
struct SongList: View {
    let songs = [
        Song(name: "怎樣"),
        Song(name: "街角的祝福")
    ]
    @State private var song: Song?
    
    var body: some View {
        NavigationSplitView {
            List(selection: $song) {
                ForEach(songs, id: \.self) { song in
                    Text(song.name)
                }
            }
            .navigationBarTitle("戴佩妮")
        } detail: {
            if let song {
                Text(song.name)
            } else {
                Text("select a song")
            }
        }
    }
}
