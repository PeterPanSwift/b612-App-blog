SwiftUI 的多段動畫  


移動沒有動畫，透明度有動畫。

struct ContentView: View {
    
    @State private var moveDistance: CGFloat = 0
    @State private var opacity: Double = 1
    
    var body: some View {
        
        VStack {
            Button("奔跑吧，Peter") {
                self.moveDistance += 100
                self.opacity -= 0.5
            }
            .font(.title)
            HStack {
                Image("peter")
                    .offset(x: moveDistance, y: 0)
                .animation(nil)
                    .opacity(opacity)
                    .animation(.linear(duration: 5))
                Spacer()
            }
        }
        
    }
    
}
