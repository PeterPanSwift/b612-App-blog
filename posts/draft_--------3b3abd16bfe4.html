製作同心圓效果


struct ContentView: View {
    let minDiameter = 50.0
    let diameterChange = 70.0
    
    var body: some View {
        ZStack {
            ForEach(0..<5) { index in
                Circle()
                    .stroke(lineWidth: 20)
                    .frame(width: minDiameter + diameterChange * Double(index))
            }
        }
        .padding()
    }
}
