One more thing: 請 AI 實現頁面的功能

最後讓我們試試整個頁面的功能都請 AI 實現。

AI 成功地完成戴佩妮歌曲內容切換的功能，而且還從 storyboard 的 Segmented Control 猜出 songs 的三首歌名，唯一的小缺點大概就是歌詞有問題。

import UIKit

class ViewController: UIViewController {
    
    // MARK: - IBOutlets
    @IBOutlet weak var songImageView: UIImageView!
    @IBOutlet weak var lyricsLabel: UILabel!
    @IBOutlet weak var songSegmentedControl: UISegmentedControl!
    @IBOutlet weak var pageControl: UIPageControl!
    
    // MARK: - Properties
    struct Song {
        let title: String
        let image: String
        let lyrics: String
    }
    
    let songs = [
        Song(title: "後果自負", 
             image: "後果自負",
             lyrics: """
                我多想學你一樣恍恍惚惚
                為愛而不得如泣如訴
                陷入 那一種幸福
                才是你想要賠上的賭注
                """),
        Song(title: "街角的祝福",
             image: "街角的祝福",
             lyrics: """
                在街角偶遇的瞬間
                時光靜止在這一刻
                你的微笑如此美麗
                讓我忘記了所有煩惱
                """),
        Song(title: "純屬意外",
             image: "純屬意外",
             lyrics: """
                這一切或許只是意外
                卻讓我無法輕易放開
                回憶中的每個畫面
                都是最美好的純真年代
                """)
    ]
    
    var currentSongIndex: Int = 0 {
        didSet {
            updateUI()
        }
    }
    
    // MARK: - Lifecycle
    override func viewDidLoad() {
        super.viewDidLoad()
        setupUI()
        updateUI()
    }
    
    // MARK: - Private Methods
    private func setupUI() {
        // 設定圖片視圖的互動性
        songImageView.isUserInteractionEnabled = true
        
        // 設定分段控制器的初始選擇
        songSegmentedControl.selectedSegmentIndex = currentSongIndex
        
        // 設定頁面控制器
        pageControl.numberOfPages = songs.count
        pageControl.currentPage = currentSongIndex
    }
    
    private func updateUI() {
        let song = songs[currentSongIndex]
        
        // 更新圖片
        songImageView.image = UIImage(named: song.image)
        
        // 更新歌詞
        lyricsLabel.text = song.lyrics
        
        // 更新分段控制器
        songSegmentedControl.selectedSegmentIndex = currentSongIndex
        
        // 更新頁面控制器
        pageControl.currentPage = currentSongIndex
        
        // 添加簡單的動畫效果
        UIView.animate(withDuration: 0.3) {
            self.songImageView.alpha = 0.5
            self.lyricsLabel.alpha = 0.5
        } completion: { _ in
            UIView.animate(withDuration: 0.3) {
                self.songImageView.alpha = 1.0
                self.lyricsLabel.alpha = 1.0
            }
        }
    }
    
    private func changeSong(to index: Int) {
        currentSongIndex = (index + songs.count) % songs.count
    }
    
    // MARK: - IBActions
    @IBAction func songSegmentChanged(_ sender: UISegmentedControl) {
        changeSong(to: sender.selectedSegmentIndex)
    }
    
    @IBAction func previousButtonTapped(_ sender: UIButton) {
        changeSong(to: currentSongIndex - 1)
    }
    
    @IBAction func nextButtonTapped(_ sender: UIButton) {
        changeSong(to: currentSongIndex + 1)
    }
    
    @IBAction func handleSwipeLeftGesture(_ sender: UISwipeGestureRecognizer) {
        // 向左滑動切換到下一首
        changeSong(to: currentSongIndex + 1)
    }
    
    @IBAction func handleSwipeRightGesture(_ sender: UISwipeGestureRecognizer) {
        // 向右滑動切換到上一首
        changeSong(to: currentSongIndex - 1)
    }
    
    
    @IBAction func pageControlValueChanged(_ sender: UIPageControl) {
        changeSong(to: sender.currentPage)
    }
}
