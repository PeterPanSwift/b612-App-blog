利用 extension 搭配 where 擴充型別功能

d

extension Array where Element == Int {
func sumInt() -> Element {
var sum = 0
for number in self {
sum = sum + number
}
return sum
}
}
[1, 2, 3].sumInt()

dd

extension Array where Element: Numeric {
func sumNumber() -> Element {
var sum: Element = 0
for number in self {
sum = sum + number
}
return sum
}
}
[1.1, 2.2].sumNumber()

可描述 generic type 或 associatedtype 的條件

可應用於 protocol extension


dd

protocol GenericProtocol {
associatedtype myType
var anyProperty: myType { get set }
}

class SomeClass: GenericProtocol {
var anyProperty = "Bob"
}

extension GenericProtocol where myType == String, Self == SomeClass {
func introduce(){
print("I'm Bob")
}
}
