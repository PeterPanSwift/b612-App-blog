從 URLRequest 或 URLSessionConfiguration 設定 HTTP header

串接第三方的 REST API 時，我們時常需要設定 HTTP header，比方設定 API key 或上傳的資料格式(Content-Type)是 JSON。

方法 1: 設定 URLRequest 的 HTTP header

針對個別的 request 設定 http header。依據 The Cat API 的文件說明，它的 HTTP header 欄位名稱是 x-api-ke，因此我們呼叫 setValue(_:forHTTPHeaderField:) 設定，DEMO-API-KEY 是它對應的內容。

let url = URL(string: "https://api.thecatapi.com/v1/favourites")!
var request = URLRequest(url: url)
request.setValue("DEMO-API-KEY", forHTTPHeaderField: "x-api-key")

除了呼叫 setValue(_:forHTTPHeaderField:)，我們可以採用以下兩種寫法:

  • addValue(_:forHTTPHeaderField:)
request.addValue("DEMO-API-KEY", forHTTPHeaderField: "x-api-key")
  • allHTTPHeaderFields
request.allHTTPHeaderFields = [
    "x-api-key": "DEMO-API-KEY"
]

方法 2: 設定 URLSessionConfiguration 的 httpAdditionalHeaders

從 URLSessionConfiguration 的 httpAdditionalHeaders 設定 http header，然後產生搭配 URLSessionConfiguration 的 URLSession，之後利用此 URLSession 發送的 request 都會帶有 httpAdditionalHeaders 設定的 http header。

let configuration = URLSessionConfiguration.default
configuration.httpAdditionalHeaders = [
    "x-api-key": "DEMO-API-KEY"
]
let urlSession = URLSession(configuration: configuration)

完整範例

方法 1: 設定 URLRequest 的 HTTP header

  • iOS 15 的寫法，搭配 async / await

假設 userId 是 123。

class CatApiClient {
    static let shared = CatApiClient()
    let baseURL = URL(string: "https://api.thecatapi.com/v1")!
    var userId = "123"
    
    func fetchFavorites() async throws -> [Favorite] {
        let baseCategoriesURL = baseURL.appendingPathComponent("favourites")
        var components = URLComponents(url: baseCategoriesURL, resolvingAgainstBaseURL: true)!
        components.queryItems = [
            URLQueryItem(name: "sub_id", value: userId)
        ]
        let categoriesURL = components.url!
        var request = URLRequest(url: categoriesURL)
        request.setValue("DEMO-API-KEY", forHTTPHeaderField: "x-api-key")
        let (data, response) = try await URLSession.shared.data(for: request)
        guard let httpResponse = response as? HTTPURLResponse, httpResponse.statusCode == 200 else {
            throw CatApiError.favoritesNotFound
        }
        let decoder = JSONDecoder()
        decoder.keyDecodingStrategy = .convertFromSnakeCase
        let favorites = try decoder.decode([Favorite].self, from: data)
        return favorites
    }
}

Task {
    do {
        let favorites = try await CatApiClient.shared.fetchFavorites()
        print(favorites)
    } catch {
        print(error)
    }
}
  • 舊版的寫法，搭配 completion handler。
class CatApiClient {
    static let shared = CatApiClient()
    let baseURL = URL(string: "https://api.thecatapi.com/v1")!
    var userId = "123"
    
    func fetchFavorites(completion: @escaping (Result<[Favorite], Error>) -> Void) {
        let baseCategoriesURL = baseURL.appendingPathComponent("favourites")
        var components = URLComponents(url: baseCategoriesURL, resolvingAgainstBaseURL: true)!
        components.queryItems = [
            URLQueryItem(name: "sub_id", value: userId)
        ]
        let categoriesURL = components.url!
        var request = URLRequest(url: categoriesURL)
        request.setValue("DEMO-API-KEY", forHTTPHeaderField: "x-api-key")
        URLSession.shared.dataTask(with: request) { data, response, error in
            if let data {
                let decoder = JSONDecoder()
                decoder.keyDecodingStrategy = .convertFromSnakeCase
                do {
                    let favorites = try decoder.decode([Favorite].self, from: data)
                    completion(.success(favorites))
                } catch {
                    completion(.failure(error))
                }
            } else if let error {
                completion(.failure(error))
            }
        }.resume()
    }
}

CatApiClient.shared.fetchFavorites { result in
    switch result {
    case .success(let favorites):
        print(favorites)
    case .failure(let error):
        print(error)
    }
}
