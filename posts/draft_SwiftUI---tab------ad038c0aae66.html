SwiftUI 的 tab 不同圖案


struct ContentView: View {
    @State private var selectTab = 0
    
    var body: some View {
        TabView(selection: $selectTab) {
            Image(systemName: "star")
                .symbolVariant(.none)
                .tag(0)
                .tabItem {
                    Label {
                        Text("星星")
                    } icon: {
                        Image(systemName: "star")
                            .symbolVariant(selectTab == 0 ? .circle : .slash)
                    }
                }
            Text("愛心")
                .tag(1)
                .tabItem {
                    Label {
                        Text("愛心")
                    } icon: {
                        Image(systemName: "heart")
                            .symbolVariant(selectTab == 1 ? .circle : .slash)
                    }
                }
        }
    }
}
