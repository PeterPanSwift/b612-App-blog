class CustomButton: UIButton {
    
    required init?(coder: NSCoder) {
        super.init(coder: coder)
        isEnabled = true 
    }
    override var isEnabled: Bool {
        didSet {
            
            backgroundColor = isEnabled ? .yellow : .gray
        }
    }
}
