parent 的 modifier 可以作用在 child 

struct ContentView: View {
    var body: some View {
        VStack {
            Text("hello")
        }
        .foregroundColor(.blue)
    }
}

struct ContentView: View {
    var body: some View {
        VStack {
            Text("hello")
                .foregroundColor(.red)
        }
        .foregroundColor(.blue)
