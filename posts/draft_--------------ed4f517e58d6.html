搭配動畫的水平滑動底線按鈕

import UIKit

class ViewController: UIViewController {
    
    @IBOutlet var barCenterXConstraints: [NSLayoutConstraint]!
    @IBOutlet var buttons: [UIButton]!
    
    override func viewDidLoad() {
        super.viewDidLoad()
        // Do any additional setup after loading the view.
    }

    
    @IBAction func tap(_ sender: UIButton) {
        let index = buttons.firstIndex(of: sender)!
        UIViewPropertyAnimator.runningPropertyAnimator(withDuration: 1, delay: 0) { [self] in
            barCenterXConstraints.forEach { $0.isActive = false }
            barCenterXConstraints[index].isActive = true
            view.layoutIfNeeded()
        }
    }
    
}
