import SwiftUI
import ReplayKit
extension UIViewController {
    static func getLastPresentedViewController() -> UIViewController? {
        let window = UIApplication.shared.windows.first {
            $0.isKeyWindow
        }
        var presentedViewController = window?.rootViewController
        while presentedViewController?.presentedViewController != nil {
            presentedViewController = presentedViewController?.presentedViewController
        }
        return presentedViewController
    }
}
struct ContentView: View {
    @State private var number = 1
    @State private var videoName: String?
func check() {
        var doc = FileManager.default.urls(for: .documentDirectory, in: .userDomainMask)[0]
        let url  = doc.appendingPathComponent("movie.mp4")
        print(FileManager.default.fileExists(atPath: url.path))
    }
    var body: some View {
        VStack {
            
            
            Text("\(number)")
                .font(.largeTitle)
                .onTapGesture {
                    number = Int.random(in: 1...100)
                    check()
                }
            Button(action: {
                RPScreenRecorder.shared().startRecording { error in
                    print(error)
                }
            }, label: {
                Text("Start Record")
            })
            
            Button(action: {
                var doc = FileManager.default.urls(for: .documentDirectory, in: .userDomainMask)[0]
                videoName = UUID().uuidString
                let url  = doc.appendingPathComponent("\(videoName!).mp4")
                print(url)
                RPScreenRecorder.shared().stopRecording(withOutput: url) { error in
                    print(error)
                }
            }, label: {
                Text("Stop Record")
            })
            
            Button(action: {
                var doc = FileManager.default.urls(for: .documentDirectory, in: .userDomainMask)[0]
                let url  = doc.appendingPathComponent("\(videoName!).mp4")
                let controller = UIActivityViewController(activityItems: [url], applicationActivities: nil)
UIViewController.getLastPresentedViewController()?.present(controller, animated: true, completion: nil)
            }, label: {
                Text("Share")
            })
        }
            
    }
}
struct ContentView_Previews: PreviewProvider {
    static var previews: some View {

    }
