使用 AI 開發英文單字的 iOS App 和 JSON 資料

開發英文單字 App 最難的地方其實不是程式，而是資料的收集，我們必須花很多時間整理單字，例句，音標等相關資訊。最近 GPT 出現後，彼得潘想到不如請 AI 幫忙整理這些英文資料，畢竟它英文超強的嘛。

GPT 果然不是省油的燈 ! 彼得潘實驗後發現只要給它單字，它就能幫忙寫出單字的中文，詞性，音標，英文例句，英文例句的翻譯。

以下我們示範四種應用，前三種請 AI 整理單字資料的 JSON，最後一種則更誇張，直接請 AI 用 SwiftUI 寫一個單字 App。

  • 將全部的單字整理成 JSON 格式，包含單字，單字的中文，詞性，音標，英文例句，英文例句的翻譯。
  • 將美國小學一年級常用的英文單字整理成 JSON 格式，包含單字，單字的中文，詞性，音標，英文例句，英文例句的翻譯。
  • 將資工系常用的英文單字整理成 JSON 格式，包含單字，單字的中文，詞性，音標，英文例句，英文例句的翻譯。

struct Word {
    var english: String
    var chinese: String
    var partOfSpeech: String
    var phonetic: String
    var sentence: String
    var sentenceTranslation: String
}
struct WordListPage: View {
    let words: [Word] = [
        Word(english: "also", chinese: "也", partOfSpeech: "adverb", phonetic: "ˈɔːlsoʊ", sentence: "I also like pizza.", sentenceTranslation: "我也喜歡比薩餅。"),
        Word(english: "animal", chinese: "動物", partOfSpeech: "noun", phonetic: "ˈænəməl", sentence: "My favorite animal is the panda.", sentenceTranslation: "我最喜歡的動物是熊貓。"),
        Word(english: "April", chinese: "四月", partOfSpeech: "noun", phonetic: "ˈeɪprəl", sentence: "My birthday is in April.", sentenceTranslation: "我的生日在四月。"),
        Word(english: "asked", chinese: "問", partOfSpeech: "verb", phonetic: "æskt", sentence: "She asked me a question.", sentenceTranslation: "她問了我一個問題。"),
        Word(english: "August", chinese: "八月", partOfSpeech: "noun", phonetic: "ɔːˈɡʌst", sentence: "We're going on vacation in August.", sentenceTranslation: "我們八月要去度假。"),
        Word(english: "books", chinese: "書籍", partOfSpeech: "noun", phonetic: "bʊks", sentence: "I love reading books.", sentenceTranslation: "我喜歡閱讀書籍。")
    ]
    
    var body: some View {
        NavigationView {
            List(words, id: \.english) { word in
                NavigationLink(destination: WordDetailPage(word: word)) {
                    Text(word.english)
                }
            }
            .navigationTitle("Words")
        }
    }
}
struct WordDetailPage: View {
    let word: Word
    
    var body: some View {
        VStack(alignment: .leading, spacing: 10) {
            Text(word.english)
                .font(.largeTitle)
            
            Text(word.chinese)
                .font(.headline)
                .foregroundColor(.gray)
            
            Divider()
            
            HStack {
                Text("Part of speech:")
                Text(word.partOfSpeech)
            }
            
            HStack {
                Text("Phonetic:")
                Text(word.phonetic)
            }
            
            Divider()
            
            Text("Example sentence:")
                .font(.headline)
                    Text(word.sentence)
        
            Text(word.sentenceTranslation)
                 .foregroundColor(.gray)
        
            Spacer()
        }
        .padding()
        .navigationBarTitleDisplayMode(.inline)
        .navigationTitle(word.english)
    }
}
@main
struct WordListApp: App {
    var body: some Scene {
        WindowGroup {
            NavigationView {
                WordListPage()
            }
        }
    }
}
