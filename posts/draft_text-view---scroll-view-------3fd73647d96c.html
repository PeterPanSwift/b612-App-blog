text view & scroll view 滑到最底部

例子: text view 增加一行文字在底下，自動滑到底部

var lyrics = ["一起長大的約定", "那樣清晰", "打過勾的我相信",
    "說好要一起旅行", "是妳如今", "唯一堅持的任性"]
    var index = 0
@IBOutlet weak var textView: UITextView!
    override func viewDidLoad() {
        super.viewDidLoad()
        // Do any additional setup after loading the view.
    }
@IBAction func add(_ sender: Any) {
        if index < lyrics.count {
            if textView.text.count > 0 {
                textView.text += "\n"
            }
            textView.text += lyrics[index]
            index += 1
            textView.scrollRangeToVisible(NSRange(location: .max, length: 0))
        }
        
    }
