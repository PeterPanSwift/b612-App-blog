利用 SwiftUI 的 DragGesture 拖曳移動元件


問題

struct DragView: View {
    @State private var offset = CGSize.zero
    var dragGesture: some Gesture {
        DragGesture()
            .onChanged({ value in
                offset = value.translation
            })
    }
    
    var body: some View {
        Image("peter")
            .offset(offset)
            .gesture(dragGesture)
    }
}

解法 1

import SwiftUI
extension CGSize {
    
    static func +(lhs: Self, rhs: Self) -> CGSize {
        CGSize(width: lhs.width + rhs.width, height: lhs.height + rhs.height)
    }
    
}

struct DragView: View {
    @State private var offset = CGSize.zero
    @State private var previousOffset = CGSize.zero
var dragGesture: some Gesture {
        DragGesture()
            .onChanged({ value in
                offset = previousOffset + value.translation
            })
            .onEnded({ value in
                previousOffset = offset
            })
    }
    
    var body: some View {
        Image("peter")
            .offset(offset)
            .gesture(dragGesture)
    }
}

解法 2

struct DragView: View {
    @State private var steadyOffset = CGSize.zero
    @State private var gestureOffset = CGSize.zero
    
    private var offset: CGSize {
        steadyOffset + gestureOffset
    }
    
    var dragGesture: some Gesture {
        DragGesture()
            .onChanged({ value in
                gestureOffset = value.translation
            })
            .onEnded({ value in
                steadyOffset = steadyOffset + gestureOffset
                gestureOffset = .zero
            })
    }
    
    var body: some View {
        Image("peter")
            .offset(offset)
            .gesture(dragGesture)
    }
}

解法 3

struct DragView: View {
    @State private var steadyOffset = CGSize.zero
    @GestureState private var gestureOffset = CGSize.zero
    
    private var offset: CGSize {
        steadyOffset + gestureOffset
    }
    
    var dragGesture: some Gesture {
        DragGesture()
            .updating($gestureOffset, body: { currentState, gestureState, _ in
                gestureState = currentState.translation
            })
            .onEnded({ value in
                steadyOffset = steadyOffset + value.translation
            })
    }
    
    var body: some View {
        Image("peter")
            .offset(offset)
            .gesture(dragGesture)
    }
}

以 @GestureState 宣告的 gestureOffset 平常只能讀取，只有在 function updating 裡可以修改。

拖曳過程中將觸發 updating，參數 currentState 是目前最新的拖曳資料，參數 gestureState 是個 inout 變數，對應 $gestureOffset。因此我們修改 gestureState 時，其實是修改 gestureOffset。

當 onEnded 觸發時，以 @GestureState 宣告的 gestureOffset 將回到初始值。
