使用 SwiftUI 的 rotation3DEffect 實現翻牌動畫(flip)

利用 SwiftUI modifier rotation3DEffect 的 3D 翻轉功能，我們可以輕易地做出撲克牌遊戲常見的翻牌動畫。

rotation3DEffect(_:axis:anchor:anchorZ:perspective:) | Apple Developer Documentation
Renders a view's content as if it's rotated in three dimensions around the specified axis.developer.apple.com

以下我們用 card back & spade ace 當牌的背面和正面說明。

翻牌通常是延著 y 軸旋轉，因此我們可以透過延著 y 軸轉 180 度實現。

.rotation3DEffect(
    .degrees(flipped ? 180 : 0),
    axis: (x: 0, y: 1, z: 0)
)

以下為翻牌動畫的完整程式

  • 版本 1: 使用 withAnimation。
struct CardView: View {
    @State private var flipped = false

    var body: some View {
        ZStack {
            Image(flipped ? .spadeAce : .cardBack)
                .resizable()
                .scaledToFill()
                .frame(width: 320, height: 450)
                .clipShape(.rect(cornerRadius: 20))
                .shadow(
                    color: .black,
                    radius: flipped ? 20 : 10,
                    y: flipped ? 20 : 10
                )
                .rotation3DEffect(
                    .degrees(flipped ? 180 : 0),
                    axis: (x: 0, y: 1, z: 0)
                )
                .onTapGesture {
                    withAnimation {
                        flipped.toggle()
                    }
                }
        }
    }
}
  • 版本 2: 使用 animation modifier。

Apple 的翻牌動畫範例

Apple 的範例比剛剛的例子複雜，由於牌的兩面呈現不一樣的元件，一面顯示 Image，一面顯示 Text，因此它使用 if else 切換顯示的內容。

struct TileView: View {
    var tile: Tile
    private let borderWidth = 5.0

    var body: some View {
        VStack {
            if tile.flipped {
                Image(systemName: "checkmark")
                    .font(Font.system(size: 120))
                    .foregroundStyle(Color.green)
                    .rotation3DEffect(.degrees(180), axis: (x: 0, y: 1, z: 0))
            } else {
                Text(tile.icon)
                    .font(Font.system(size: 80))
                Text(tile.word)
                    .font(.title)
            }
        }
        .frame(width: Tile.size - borderWidth * 2, height: Tile.size - borderWidth * 2)
        .overlay(RoundedRectangle(cornerRadius: 16).stroke(borderStyle(), lineWidth: borderWidth))
        .background(Color.purple.opacity(0.5))
        .clipShape(RoundedRectangle(cornerRadius: 16))
        .rotation3DEffect(.degrees(tile.flipped ? 180 : 0), axis: (x: 0, y: 1, z: 0))
        .animation(.default, value: tile.flipped)
    }

    func borderStyle() -> some ShapeStyle {
        return LinearGradient(colors: [
            Color(red: 157.0/255.0, green: 153.0/255.0, blue: 244.0/255.0),
            Color(red: 246.0/255.0, green: 206.0/255.0, blue: 241.0/255.0),
        ], startPoint: .topLeading, endPoint: .bottomTrailing)
    }
}

AI 實現更酷炫的翻牌動畫

包含了翻轉、縮放、發光、粒子效果。

struct CardView: View {
    @State private var flipped = false
    @State private var animated3d = false
    @State private var scale = 1.0
    @State private var rotation = 0.0
    @State private var offset = CGSize.zero

    var body: some View {
        ZStack {
            Image(flipped ? .spadeAce : .cardBack)
                .resizable()
                .scaledToFill()
                .frame(width: 320, height: 450)
                .clipShape(.rect(cornerRadius: 20))
                .shadow(
                    color: .black.opacity(0.2),
                    radius: flipped ? 20 : 10,
                    y: flipped ? 20 : 10
                )
                .rotation3DEffect(
                    .degrees(animated3d ? 180 : 0),
                    axis: (x: 0, y: 1, z: 0)
                )
                .rotationEffect(.degrees(rotation))
                .scaleEffect(scale)
                .offset(offset)
                .blur(radius: animated3d ? 0 : 0.5)
                .animation(
                    .spring(
                        response: 0.6,
                        dampingFraction: 0.8,
                        blendDuration: 0
                    ),
                    value: animated3d
                )
                .onTapGesture {
                    withAnimation {
                        scale = 0.9
                        rotation = flipped ? -5 : 5
                        offset = CGSize(
                            width: flipped ? -10 : 10,
                            height: 0
                        )
                        Task {
                            try await Task.sleep(for: .seconds(0.1))
                            animated3d.toggle()
                            flipped.toggle()
                            try await Task.sleep(for: .seconds(0.3))
                            withAnimation(.spring()) {
                                scale = 1
                                rotation = 0
                                offset = .zero
                            }
                        }
                    }
                }
        }
        .glow(color: flipped ? .red : .blue, radius: 20)
        .modifier(ParticleEffect(isActive: flipped))
    }
}

struct ParticleEffect: ViewModifier {
    let isActive: Bool

    func body(content: Content) -> some View {
        content
            .overlay(
                ZStack {
                    if isActive {
                        ForEach(0..<20) { _ in
                            Circle()
                                .fill(.white.opacity(0.5))
                                .frame(width: 8, height: 8)
                                .modifier(ParticleMotionModifier())
                        }
                    }
                }
            )
    }
}

struct ParticleMotionModifier: ViewModifier {
    @State private var isAnimating = false

    func body(content: Content) -> some View {
        content
            .offset(
                x: isAnimating ? CGFloat.random(in: -100...100) : 0,
                y: isAnimating ? CGFloat.random(in: -100...100) : 0
            )
            .opacity(isAnimating ? 0 : 1)
            .onAppear {
                withAnimation(
                    .easeOut(duration: Double.random(in: 0.5...1.0))
                ) {
                    isAnimating = true
                }
            }
    }
}

extension View {
    func glow(color: Color, radius: CGFloat) -> some View {
        self
            .shadow(color: color.opacity(0.3), radius: radius / 3)
            .shadow(color: color.opacity(0.2), radius: radius / 2)
            .shadow(color: color.opacity(0.1), radius: radius)
    }
}
