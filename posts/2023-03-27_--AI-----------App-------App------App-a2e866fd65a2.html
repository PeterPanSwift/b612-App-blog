AI 的其中一個強項是模仿跟舉一反三，因此如果已經有一份寫好的程式，我們可以請它幫忙修改,，快速生成顯示新主題但維持同樣架構的全新 App。

Apple’s Meal Tracker App

class FoodTableViewController: UITableViewController {
    
    var meals: [Meal] {
        let firstBreakfastFood = Food(name: "Eggs", description: "Scrambled with bacon in a frying pan.")
        let secondBreakfastFood = Food(name: "Hashbrowns", description: "Cut potatoes then fry in oil until brown")
        let thirdBreakfastFood = Food(name: "Bacon", description: "Key food in all breakfast meals.")
        let breakfast = Meal(name: "Breakfast", food: [firstBreakfastFood, secondBreakfastFood, thirdBreakfastFood])
        
        let firstLunchFood = Food(name: "Sandwich", description: "Easy to fix. Always delicious")
        let secondLunchFood = Food(name: "Chips", description: "Put a few in the sandwhich for enhanced flavor")
        let thirdLunchFood = Food(name: "Apple", description: "An apple a day keeps the doctor away.")
        let lunch = Meal(name: "Lunch", food: [firstLunchFood, secondLunchFood, thirdLunchFood])
        
        let firstDinnerFood = Food(name: "Steak", description: "Everyone needs some good protein.")
        let secondDinnerFood = Food(name: "Potatoes", description: "Great addition to go along with steak.")
        let thirdDinnerFood = Food(name: "Brocolli", description: "Always finish out the food pyramid.")
        let dinner = Meal(name: "Dinner", food: [firstDinnerFood, secondDinnerFood, thirdDinnerFood])
        
        return [breakfast, lunch, dinner]
    }

    // MARK: - Table view data source

    override func numberOfSections(in tableView: UITableView) -> Int {
        return meals.count
    }

    override func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        let meal = meals[section]
        let food = meal.food
        
        return food.count
    }
    
    override func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "Food", for: indexPath)
        let meal = meals[indexPath.section]
        let food = meal.food[indexPath.row]
        
        // Configure the cell...
        var content = cell.defaultContentConfiguration()
        content.text = food.name
        content.secondaryText = food.description
        cell.contentConfiguration = content
        
        return cell
    }
    
    override func tableView(_ tableView: UITableView, titleForHeaderInSection section: Int) -> String? {
        let meal = meals[section]
        
        return meal.name
    }
    
}

請 GPT 4 將美食 App 變成電影 App

  • 問題

將表格顯示的資料改成顯示台灣繁體中文的電影，第一個 section 是喜劇，第二個 section 是愛情片，第三個 section 是恐怖片。

  • 回答
class MovieTableViewController: UITableViewController {
    
    struct Movie {
        let name: String
        let description: String
    }
    
    struct MovieCategory {
        let name: String
        let movies: [Movie]
    }
    
    var movieCategories: [MovieCategory] {
        let comedies = MovieCategory(name: "喜劇", movies: [
            Movie(name: "逗陣冤家", description: "兩個性格截然不同的鄰居互相捉弄，引發一連串的笑料。"),
            Movie(name: "瘋狂假期", description: "一家人計畫度假，卻遇到種種瘋狂的困境。"),
            Movie(name: "賭神", description: "賭博高手與新手攜手挑戰黑幫，場面十分搞笑。")
        ])
        
        let romance = MovieCategory(name: "愛情片", movies: [
            Movie(name: "那些年，我們一起追的女孩", description: "一群男生為了追求理想中的女生，經歷了青春期的成長與愛情的酸甜苦辣。"),
            Movie(name: "不能說的秘密", description: "天才鋼琴家與謎樣女生的愛情故事，充滿濃厚的浪漫氛圍。"),
            Movie(name: "愛情起跑點", description: "一對年輕戀人，克服種種困難，努力維持愛情。")
        ])
        
        let horror = MovieCategory(name: "恐怖片", movies: [
            Movie(name: "詭絲", description: "詭異的絲線纏繞著一個小鎮，引發一連串恐怖的事件。"),
            Movie(name: "鬼打牆", description: "一群年輕人參觀廢棄鬼屋，卻無法擺脫無盡的惡夢。"),
            Movie(name: "厲陰宅", description: "一家人搬進新屋，卻不知新屋內住著邪靈，開始經歷恐怖的詭異事件。")
        ])
        
        return [comedies, romance, horror]
    }
    
    // MARK: - Table view data source
    
    override func numberOfSections(in tableView: UITableView) -> Int {
        return movieCategories.count
    }
    
    override func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        let movieCategory = movieCategories[section]
        let movies = movieCategory.movies
        
        return movies.count
    }
    
    override func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "Movie", for: indexPath)
        let movieCategory = movieCategories[indexPath.section]
        let movie = movieCategory.movies[indexPath.row]
        
        // Configure the cell...
        var content = cell.defaultContentConfiguration()
        content.text = movie.name
        content.secondaryText = movie.description
        cell.contentConfiguration = content
        
        return cell
    }
    
    override func tableView(_ tableView: UITableView, titleForHeaderInSection section: Int) -> String? {
        let movieCategory = movieCategories[section]
        
        return movieCategory.name
    }
    
}
