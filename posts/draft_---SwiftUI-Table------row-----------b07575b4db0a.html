利用 SwiftUI Table 讓表格的 row 有多個可點選的區塊

iOS App 上


點選不同區塊跳到不同頁面

點選觸發不同功能

隱藏向右的箭頭


struct PostTable: View {
    let posts = [        Post(author: "John", content: "Hello, World!"),        Post(author: "Mary", content: "Hello, World!"),    ]
    
    var body: some View {
        NavigationStack {
            Table(posts) {
                TableColumn("post") { post in
                    NavigationLink(value: post) {
                        VStack(alignment: .leading) {
                            NavigationLink(value: post.author) {
                                Text(post.author)
                            }
                            Text(post.content)
                            Button {
                                print("heart")
                            } label: {
                                Image(systemName: "heart")
                            }
                        }
                    }
                }
                
            }
            .navigationDestination(for: Post.self) { post in
                Text(post.content)
            }
            .navigationDestination(for: String.self) { author in
                Text(author)
            }
        }
    }
}
