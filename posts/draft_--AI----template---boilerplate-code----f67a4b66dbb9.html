產生 SwiftUI LazyVGrid 的 template

import SwiftUI

struct LazyVGridTemplate: View {
    let columns = [
        GridItem(.flexible()),
        GridItem(.flexible())
    ]

    let items = Array(1...20).map { "Item \($0)" }

    var body: some View {
        ScrollView {
            LazyVGrid(columns: columns, spacing: 20) {
                ForEach(items, id: \.self) { item in
                    Text(item)
                        .frame(maxWidth: .infinity)
                        .padding()
                        .background(.blue)
                        .clipShape(.rect(cornerRadius: 8))
                        .foregroundStyle(.white)
                }
            }
            .padding(.horizontal)
        }
    }
}
