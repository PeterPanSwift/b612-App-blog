請 AI 寫 UI test


好的,以下是針對你提供的 iOS App 所寫的 UI 測試程式碼:

```swift
import XCTest

class GuessNumberGameUITests: XCTestCase {
    
    var app: XCUIApplication!
    
    override func setUp() {
        super.setUp()
        continueAfterFailure = false
        app = XCUIApplication()
        app.launch()
    }
    
    func testGuessNumberGameUI() {
        let numberTextField = app.textFields["numberTextField"]
        let guessButton = app.buttons["guessButton"]
        let resultLabel = app.staticTexts["resultLabel"]
        let guessWrongCountLabel = app.staticTexts["guessWrongCountLabel"]
        let gameOverLabel = app.staticTexts["gameOverLabel"]
        let replayButton = app.buttons["replayButton"]
        
        // 檢查初始 UI 元件狀態
        XCTAssertEqual(resultLabel.label, "結果")
        XCTAssertEqual(guessWrongCountLabel.label, "猜錯0次")
        XCTAssertTrue(gameOverLabel.exists)
        XCTAssertFalse(gameOverLabel.isHittable)
        XCTAssertFalse(replayButton.isEnabled)
        
        // 測試猜對數字
        numberTextField.tap()
        numberTextField.typeText("50")
        guessButton.tap()
        
        if resultLabel.label == "準!" {
            XCTAssertFalse(guessButton.isEnabled)
            XCTAssertTrue(replayButton.isEnabled)
        } else {
            // 測試猜錯 3 次
            for _ in 1...2 {
                numberTextField.tap()
                numberTextField.typeText("60")
                guessButton.tap()
            }
            
            XCTAssertEqual(guessWrongCountLabel.label, "猜錯3次")
            XCTAssertTrue(gameOverLabel.isHittable)
            XCTAssertFalse(guessButton.isEnabled)
            XCTAssertTrue(replayButton.isEnabled)
        }
        
        // 測試重玩功能
        replayButton.tap()
        XCTAssertEqual(guessWrongCountLabel.label, "猜錯0次")
        XCTAssertFalse(gameOverLabel.isHittable)
        XCTAssertTrue(guessButton.isEnabled)
        XCTAssertFalse(replayButton.isEnabled)
    }
    
    override func tearDown() {
        app = nil
        super.tearDown()
    }
}
```

這個 UI 測試程式碼包含以下測試步驟:

1. 在 `setUp()` 方法中初始化 `XCUIApplication` 並啟動應用程式。

2. 在 `testGuessNumberGameUI()` 方法中:
   - 檢查初始 UI 元件的狀態。
   - 測試猜對數字的情況。
   - 如果沒猜對,則測試猜錯 3 次的情況。
   - 測試重玩功能。

3. 在 `tearDown()` 方法中清理資源。

這個 UI 測試會模擬使用者與應用程式的互動,並驗證 UI 元件的狀態變化是否符合預期。你可以根據需要進一步擴展或修改測試案例。

請注意,這個測試程式碼假設你已經在 Xcode 中設置了 UI 測試 Target,並且在 Storyboard 中為相關的 UI 元件設置了 Accessibility Identifier。
