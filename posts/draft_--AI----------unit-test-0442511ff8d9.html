當然,我來為你的 ViewController 寫單元測試。以下是針對 ViewController 的單元測試程式碼:

```swift
import XCTest
@testable import Demo

class ViewControllerTests: XCTestCase {
    var viewController: ViewController!
    
    override func setUp() {
        super.setUp()
        let storyboard = UIStoryboard(name: "Main", bundle: nil)
        viewController = storyboard.instantiateViewController(withIdentifier: "ViewController") as? ViewController
        viewController.loadViewIfNeeded()
    }
    
    override func tearDown() {
        viewController = nil
        super.tearDown()
    }
    
    func testInitialState() {
        XCTAssertEqual(viewController.guessWrongCount, 0)
        XCTAssertEqual(viewController.guessWrongCountLabel.text, "猜錯0次")
        XCTAssertEqual(viewController.resultLabel.text, "結果")
        XCTAssertTrue(viewController.guessButton.isEnabled)
        XCTAssertFalse(viewController.replayButton.isEnabled)
        XCTAssertTrue(viewController.gameOverLabel.isHidden)
    }
    
    func testGuessCorrectNumber() {
        viewController.answer = 50
        viewController.numberTextField.text = "50"
        viewController.guess(self)
        
        XCTAssertEqual(viewController.resultLabel.text, "準!")
        XCTAssertFalse(viewController.guessButton.isEnabled)
        XCTAssertTrue(viewController.replayButton.isEnabled)
    }
    
    func testGuessWrongNumberTooSmall() {
        viewController.answer = 50
        viewController.numberTextField.text = "30"
        viewController.guess(self)
        
        XCTAssertEqual(viewController.guessWrongCount, 1)
        XCTAssertEqual(viewController.guessWrongCountLabel.text, "猜錯1次")
        XCTAssertEqual(viewController.resultLabel.text, "太小了")
        XCTAssertEqual(viewController.numberTextField.text, "")
    }
    
    func testGuessWrongNumberTooBig() {
        viewController.answer = 50
        viewController.numberTextField.text = "80"
        viewController.guess(self)
        
        XCTAssertEqual(viewController.guessWrongCount, 1)
        XCTAssertEqual(viewController.guessWrongCountLabel.text, "猜錯1次")
        XCTAssertEqual(viewController.resultLabel.text, "太大了")
        XCTAssertEqual(viewController.numberTextField.text, "")
    }
    
    func testGuessWrong3Times() {
        viewController.answer = 50
        viewController.numberTextField.text = "30"
        viewController.guess(self)
        viewController.numberTextField.text = "80"
        viewController.guess(self)
        viewController.numberTextField.text = "20"
        viewController.guess(self)
        
        XCTAssertEqual(viewController.guessWrongCount, 3)
        XCTAssertEqual(viewController.guessWrongCountLabel.text, "猜錯3次")
        XCTAssertFalse(viewController.gameOverLabel.isHidden)
        XCTAssertTrue(viewController.replayButton.isEnabled)
        XCTAssertFalse(viewController.guessButton.isEnabled)
    }
    
    func testReplayGame() {
        viewController.guessWrongCount = 3
        viewController.guessWrongCountLabel.text = "猜錯3次"
        viewController.gameOverLabel.isHidden = false
        viewController.resultLabel.text = "太大了"
        viewController.guessButton.isEnabled = false
        viewController.numberTextField.text = "80"
        
        viewController.replay(self)
        
        XCTAssertEqual(viewController.guessWrongCount, 0)
        XCTAssertEqual(viewController.guessWrongCountLabel.text, "猜錯0次")
        XCTAssertTrue(viewController.gameOverLabel.isHidden)
        XCTAssertEqual(viewController.resultLabel.text, "結果")
        XCTAssertTrue(viewController.guessButton.isEnabled)
        XCTAssertEqual(viewController.numberTextField.text, "")
    }
}
```

這個單元測試涵蓋了以下幾個場景:

1. 測試初始狀態: 確保在視圖控制器載入時,各個 UI 元件的狀態正確。
2. 測試猜對數字: 當猜對數字時,結果標籤應顯示 "準!",猜測按鈕應被禁用,重玩按鈕應啟用。
3. 測試猜錯數字(太小): 當猜測的數字太小時,猜錯次數應增加,顯示相應的猜錯次數,結果標籤顯示 "太小了",並清空輸入欄位。
4. 測試猜錯數字(太大): 當猜測的數字太大時,猜錯次數應增加,顯示相應的猜錯次數,結果標籤顯示 "太大了",並清空輸入欄位。
5. 測試猜錯3次: 當猜錯3次時,顯示遊戲結束標籤,啟用重玩按鈕,禁用猜測按鈕。
6. 測試重新開始遊戲: 當點擊重玩按鈕時,重置各個 UI 元件的狀態,隱藏遊戲結束標籤,清空輸入欄位,啟用猜測按鈕。

你可以在專案中新增一個名為 ViewControllerTests.swift 的檔案,然後將上述程式碼貼入其中。確保在測試目標(test target)中包含這個檔案。

這樣就完成了對 ViewController 的單元測試。記得在實際執行測試之前,先建置專案並確保主程式碼沒有錯誤。然後你可以使用 Xcode 的測試導覽器(Test Navigator)來執行這些測試並查看結果。
