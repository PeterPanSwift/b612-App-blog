判斷剪刀石頭布勝負的多種方法

剪刀石頭布是許多初學者學程式都做過的題目，通常我們會用 enum 定義 GameResult 和 Sign，然後透過 if else 或 switch 比較勝負。

enum GameResult {
    case win, lose, draw
}
enum Sign {
    case rock
    case paper
    case scissors
}

除了使用基本的方法比較，我們也可以結合程式語言的某些功能，實現更精簡的寫法。

以下我們介紹六種判斷剪刀石頭布勝負的方法。(ps: 感謝 AI 建議其中的幾種方法)

  • 基本的 if else 寫法。
  • 基本的 switch 寫法。
  • 使用 Tuple。
  • 使用 Dictionary。
  • 使用 computed property。
  • 使用 CaseIterable 和索引。

基本的 if else 寫法

enum Sign {
    case rock
    case paper
    case scissors
    
    func gameResult(against opponentSign: Sign) -> GameResult {
        if self == opponentSign {
            return .draw
        }
        if (self == .rock && opponentSign == .scissors) ||
           (self == .paper && opponentSign == .rock) ||
           (self == .scissors && opponentSign == .paper) {
            return .win
        } else {
            return .lose
        }
    }
}

基本的 switch 寫法

enum Sign {
    case rock
    case paper
    case scissors
    
    func gameResult(against opponentSign: Sign) -> GameResult {
        if self == opponentSign {
            return .draw
        }
        switch self {
        case .rock:
            if opponentSign == .scissors {
                return .win
            }
        case .paper:
            if opponentSign == .rock {
                return .win
            }
        case .scissors:
            if opponentSign == .paper {
                return .win
            }
        }
        return .lose
    }
    
}

使用 Tuple

  • 寫法 1。
enum Sign {
    case rock
    case paper
    case scissors
    
    func gameResult(against opponentSign: Sign) -> GameResult {
        if self == opponentSign { return .draw }
        switch (self, opponentSign) {
        case (.rock, .scissors), (.paper, .rock), (.scissors, .paper):
            return .win
        default:
            return .lose
        }
    }
}
  • 寫法 2。
enum Sign {
    case rock
    case paper
    case scissors

    func gameResult(against opponentSign: Sign) -> GameResult {
        switch (self, opponentSign) {
        case (.rock, .scissors), (.paper, .rock), (.scissors, .paper):
            return .win
        case (.rock, .paper), (.paper, .scissors), (.scissors, .rock):
            return .lose
        default:
            return .draw
        }
    }
}

使用 Dictionary

enum Sign {
    case rock
    case paper
    case scissors
    
    static let winningMoves: [Sign: Sign] = [
        .rock: .scissors,
        .paper: .rock,
        .scissors: .paper
    ]
    
    func gameResult(against opponentSign: Sign) -> GameResult {
        if self == opponentSign {
            return .draw
        }
        return Sign.winningMoves[self] == opponentSign ? .win : .lose
    }
}

使用 computed property

enum Sign {
    case rock
    case paper
    case scissors
    
    var beats: Sign {
        switch self {
        case .rock:
            return .scissors
        case .paper:
            return .rock
        case .scissors:
            return .paper
        }
    }
    
    func gameResult(against opponentSign: Sign) -> GameResult {
        if self == opponentSign {
            return .draw
        }
        return self.beats == opponentSign ? .win : .lose
    }
}

使用 CaseIterable 和索引

enum Sign: CaseIterable, Equatable {
    case rock
    case paper
    case scissors

    func gameResult(against opponentSign: Sign) -> GameResult {
        let signs = Sign.allCases
        if self == opponentSign {
            return .draw
        } else if signs[(signs.firstIndex(of: self)! + 1) % signs.count] == opponentSign {
            return .lose
        } else {
            return .win
        }
    }
}
