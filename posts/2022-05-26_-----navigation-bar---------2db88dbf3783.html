從 Interface Builder & 程式設定 UIKit navigation bar & tab bar 的樣式
從 iOS 13 開始，navigation bar & tab bar 的樣式可透過各種 appearance 設定，比方…medium.com

不過最近彼得潘測試動態改變 bar 的背景顏色，比方點選 button 讓 bar 的背景顏色在 yellow / green 間切換時，發現在 iOS 14 使用 UINavigationBarAppearance 設定會有問題。

因此若想動態改變背景顏色，建議在 iOS 15 以上使用 UINavigationBarAppearance，iOS 14 之前使用 barTintColor。例如以下例子:

class ViewController: UIViewController {
    
    var isYellow = true
    override func viewDidLoad() {
        super.viewDidLoad()
        
        updateNavBarColor()
    }
    func updateNavBarColor() {
        
        if #available(iOS 15.0, *) {
            let barAppearance = UINavigationBarAppearance()
            barAppearance.configureWithOpaqueBackground()
            barAppearance.backgroundColor = isYellow ? .yellow : .green
                    navigationController?.navigationBar.standardAppearance = barAppearance
            navigationController?.navigationBar.scrollEdgeAppearance = barAppearance
            navigationController?.navigationBar.compactAppearance = barAppearance
            navigationController?.navigationBar.compactScrollEdgeAppearance = barAppearance
        } else {
            navigationController?.navigationBar.barTintColor = isYellow ? .yellow : .green
        }
    }
    
    @IBAction func tap(_ sender: Any) {
        isYellow.toggle()
        updateNavBarColor()
      
    }
    
}
