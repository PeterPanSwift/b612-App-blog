用 dump 或 Mirror 印出 class / struct 每個 property 的內容

用 dump 或 Mirror 印出 class / struct 每個 property 的內容




class Baby {
    var name = "peter"
    var age = 1
    
    init(name: String = "peter", age: Int = 1) {
        self.name = name
        self.age = age
    }
}


let babies = [
    Baby(name: "peter", age: 1),
    Baby(name: "john", age: 2),
    Baby(name: "mary", age: 3)
]

dump(babies)


func printProperties(of object: Any) {
    let mirror = Mirror(reflecting: object)
    
    print("Properties of \(mirror.subjectType):")
    for child in mirror.children {
        if let propertyName = child.label {
            print("\(propertyName): \(child.value)")
        }
    }
}
