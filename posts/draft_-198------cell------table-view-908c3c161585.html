#198 搭配內建 cell 樣式顯示 table view

基本功能

  • 將字串的 array 顯示在一個 section 的 table view。

範例: 文藝復興時期的藝術家。

  • 字串 array 可手動輸入或請 ChatGPT 生成。

  •  cell 使用 Basic Style。

進階功能

  • 有多個字串的 array，將它們顯示在多個 section 的 table view。

範例: 浪漫主義 & 維多利亞時代的英國作家。

解答參考

  • 一個 section。
class ViewController: UIViewController {
    @IBOutlet weak var tableView: UITableView!
    let renaissanceArtists = ["達芬奇", "拉斐爾", "米開朗基羅", "倫勃朗", "波提切利", "維米爾", "布魯克斯", "格奧爾基", "馬賽克", "阿瑪爾托"]
    
    override func viewDidLoad() {
        super.viewDidLoad()
        tableView.dataSource = self
    }
}

extension ViewController: UITableViewDelegate, UITableViewDataSource {
    
    func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        return renaissanceArtists.count
    }
    
    func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "Cell", for: indexPath)
        cell.textLabel?.text = renaissanceArtists[indexPath.row]
        return cell
    }    
}
  • 多個 section。
class ViewController: UIViewController {
    @IBOutlet weak var tableView: UITableView!
    let romanticWriters = ["雪萊", "拜倫", "濟慈"]
    let victorianWriters = ["狄更斯", "王爾德", "蕭伯納"]
    
    override func viewDidLoad() {
        super.viewDidLoad()
        tableView.dataSource = self
    }
}

extension ViewController: UITableViewDelegate, UITableViewDataSource {
    
    func numberOfSections(in tableView: UITableView) -> Int {
        return 2
    }
    
    func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        if section == 0 {
            return romanticWriters.count
        } else {
            return victorianWriters.count
        }
    }
    
    func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "Cell", for: indexPath)
        if indexPath.section == 0 {
            cell.textLabel?.text = romanticWriters[indexPath.row]
        } else {
            cell.textLabel?.text = victorianWriters[indexPath.row]
        }
        return cell
    }
    
    func tableView(_ tableView: UITableView, titleForHeaderInSection section: Int) -> String? {
        if section == 0 {
            return "浪漫主義"
        } else {
            return "維多利亞時代"
        }
    }
}
