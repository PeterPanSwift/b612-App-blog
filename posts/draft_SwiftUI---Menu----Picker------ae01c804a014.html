SwiftUI 的 Menu 搭配 Picker 實現選單

SwiftUI 的 Menu 搭配 Picker 實現選單


struct ContentView: View {
    let pres = [" ", "a", "an"]
    @State private var selectPre = "a"
var body: some View {
Menu {
            Picker(selection: $selectPre) {
                ForEach(pres, id: \.self) { pre in
                    Text(pre)
                        .font(.largeTitle)
                        .foregroundColor(.blue)
                }
            } label: {
                Text("pre")
            }
        } label: {
            Text(selectPre)
                .font(.system(size: 40))
                .frame(width: 50)
                .background(.yellow)
        }
    }
}
