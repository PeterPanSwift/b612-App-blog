override func viewDidLoad() {
   super.viewDidLoad()
   let flowLayout = collectionView.collectionViewLayout as? UICollectionViewFlowLayout
   let width = (view.frame.width - 10 * 3) / 4
   flowLayout?.itemSize = CGSize(width: width, height: width)
