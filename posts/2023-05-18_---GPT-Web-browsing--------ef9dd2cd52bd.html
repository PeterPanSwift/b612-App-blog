定義以下 API 回傳的 JSON 對應的 Codable 型別
https://rss.applemarketingtools.com/api/v2/us/apps/top-free/10/apps.json

import Foundation

struct Feed: Codable {
    let feed: FeedDetails
}

struct FeedDetails: Codable {
    let title: String
    let id: String
    let author: Author
    let links: [Link]
    let copyright: String
    let country: String
    let icon: String
    let updated: String
    let results: [App]
}

struct Author: Codable {
    let name: String
    let url: String
}

struct Link: Codable {
    let selfLink: String
    
    enum CodingKeys: String, CodingKey {
        case selfLink = "self"
    }
}

struct App: Codable {
    let artistName: String
    let id: String
    let name: String
    let releaseDate: String
    let kind: String
    let artworkUrl100: String
    let genres: [String]
    let url: String
}

用 SwiftUI，JSONDecoder & AsyncImage 寫一個 App 清單畫面，資料來自以下網址，包含 App 名字和縮圖。 https://rss.applemarketingtools.com/api/v2/us/apps/top-free/10/apps.json

GPT 定義的資料型別名稱 App 會跟內建的 App protocol 衝突，因此以下我們將它改成 AppInfo。

  • TopApps.swift
import Foundation

struct Feed: Codable {
    let results: [AppInfo]
}

struct AppInfo: Codable, Identifiable {
    let id: String
    let artistName: String
    let name: String
    let artworkUrl100: URL
}

struct TopApps: Codable {
    let feed: Feed
}
  • ViewModel.swift
import SwiftUI

class ViewModel: ObservableObject {
    @Published var apps: [AppInfo] = []
    
    func fetchApps() {
        guard let url = URL(string: "https://rss.applemarketingtools.com/api/v2/us/apps/top-free/10/apps.json") else { return }
        URLSession.shared.dataTask(with: url) { data, response, error in
            if let data = data {
                do {
                    let topApps = try JSONDecoder().decode(TopApps.self, from: data)
                    DispatchQueue.main.async {
                        self.apps = topApps.feed.results
                    }
                } catch {
                    print("Failed to decode JSON: \(error)")
                }
            } else if let error = error {
                print("Failed to fetch data: \(error)")
            }
        }.resume()
    }
}
  • ContentView.swift
import SwiftUI

struct ContentView: View {
    @StateObject var viewModel = ViewModel()
    
    var body: some View {
        NavigationView {
            List(viewModel.apps) { app in
                HStack {
                    AsyncImage(url: app.artworkUrl100) { image in
                        image.resizable().frame(width: 50, height: 50)
                    } placeholder: {
                        ProgressView()
                    }
                    VStack(alignment: .leading) {
                        Text(app.name)
                            .font(.headline)
                        Text(app.artistName)
                            .font(.subheadline)
                    }
                }
            }
            .navigationTitle("Top Free Apps")
        }
        .onAppear {
            viewModel.fetchApps()
        }
    }
}

// 引入所需的模組
import Charts     // 用於繪製圖表
import SwiftUI    // 用於建立使用者介面
import FoodTruckKit // 這應該是一個自定義的模組，可能包含了和食物相關的數據模型和邏輯
import AppIntents // 用於與 Siri 進行互動

// 定義一個 SwiftUI View，名為 TopFiveDonutsView
struct TopFiveDonutsView: View {
    // 這個 View 觀察一個 FoodTruckModel 物件
    @ObservedObject var model: FoodTruckModel

    // timeframe 用來表示時間範圍，初始設為一週
    @State private var timeframe: Timeframe = .week

    // today 是當前日期的開始時間
    let today = Calendar.current.startOfDay(for: .now)

    // body 是 View 的主體，定義了界面的布局和行為
    var body: some View {
        // 使用 ScrollView 包裹所有元素，當內容超出視圖範圍時，可以滾動查看
        ScrollView {
            // VStack 是垂直堆疊的 View
            VStack(alignment: .leading) {
                // Picker 允許使用者選擇時間範圍
                Picker("Timeframe", selection: $timeframe) {
                    Text("Day").tag(Timeframe.today)
                    Text("Week").tag(Timeframe.week)
                    Text("Month").tag(Timeframe.month)
                    Text("Year").tag(Timeframe.year)
                }
                .pickerStyle(.segmented)
                .padding(.bottom, 10)

                // 顯示 TopFiveDonutsChart，這個 View 顯示售出最多的五種甜甜圈的圖表
                TopFiveDonutsChart(model: model, timeframe: timeframe)

                // 在 iOS 上，SiriTipView 會顯示一些與 Siri 相關的提示
                #if os(iOS)
                SiriTipView(intent: ShowTopDonutsIntent(timeframe: timeframe))
                    .padding(.top, 20)
                #endif
            }
            .padding()
        }
        .navigationTitle("Top 5 Donuts")
        .background()
    }
}
