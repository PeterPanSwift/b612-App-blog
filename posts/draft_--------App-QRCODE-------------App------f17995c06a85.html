struct QRCodeView: View {
    @State private var brightness: CGFloat?
    let scene = {
        UIApplication.shared.connectedScenes
              .filter { $0.activationState == .foregroundActive }
              .first { $0 is UIWindowScene } as? UIWindowScene
    }()
    
   
    var body: some View {
        ZStack {
            Color.white
                .ignoresSafeArea()
            Image("QRCode")
        }
        .onAppear {
            if brightness == nil {
                brightness = scene?.screen.brightness
            }
            scene?.screen.brightness = 1
        }
        .onDisappear {
            if let brightness {
                scene?.screen.brightness = brightness
            }
        }
            
    }
}
