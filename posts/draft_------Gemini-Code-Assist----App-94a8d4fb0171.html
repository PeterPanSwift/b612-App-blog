import SwiftUI
import Combine

struct Mole: Identifiable {
    let id = UUID()
    var isVisible: Bool = false
    var position: Int
}

struct ContentView: View {
    @State private var moles: [Mole] = []
    @State private var score = 0
    @State private var timer = Timer.publish(every: 1, on: .main, in: .common).autoconnect() //timer for each mole appear
    @State private var gameTimer = Timer.publish(every: 1, on: .main, in: .common).autoconnect() // timer for game time
    @State private var timeLeft = 30 // Time limit in seconds
    @State private var isGameStarted = false
    @State private var difficulty = 1.0 // control the speed of mole appear. bigger number means faster.

    let gridColumns = [GridItem(.flexible()), GridItem(.flexible()), GridItem(.flexible())]

    var body: some View {
        VStack {
            Text("Score: \(score)")
                .font(.title)
            Text("Time: \(timeLeft)")
                .font(.title2)

            if !isGameStarted {
                Button("Start Game") {
                    startGame()
                }
                .font(.title2)
                .padding()
                Button("Easy") {
                    difficulty = 1.5
                }
                .font(.title2)
                .padding()
                Button("Normal") {
                    difficulty = 1.0
                }
                .font(.title2)
                .padding()
                Button("Hard") {
                    difficulty = 0.5
                }
                .font(.title2)
                .padding()

            } else {
                LazyVGrid(columns: gridColumns) {
                    ForEach(moles) { mole in
                        MoleView(isVisible: mole.isVisible, onMoleTapped: {
                            whackMole(at: mole.position)
                        })
                        .id(mole.id)
                        .frame(width: 100, height: 100)
                    }
                }
                .padding()
                .onReceive(timer) { _ in
                    moleTurn()
                }
                .onReceive(gameTimer) { _ in
                    gameCountDown()
                }
            }
        }
        .padding()
        .onAppear {
            initializeMoles()
        }
        .alert(isPresented: .constant(timeLeft == 0)) {
            Alert(title: Text("Game Over!"), message: Text("Your final score is \(score)"), dismissButton: .default(Text("Play Again")){
                restartGame()
            })
        }
    }

    func initializeMoles() {
        for i in 0..<9 {
            moles.append(Mole(position: i))
        }
    }

    func startGame() {
        isGameStarted = true
        timeLeft = 30
        score = 0

        //restart timer
        timer = Timer.publish(every: difficulty, on: .main, in: .common).autoconnect()
        gameTimer = Timer.publish(every: 1, on: .main, in: .common).autoconnect()
    }
    
    func restartGame(){
        isGameStarted = false
        initializeMoles()
    }

    func moleTurn() {
        // Hide all moles
        for i in moles.indices {
            moles[i].isVisible = false
        }

        // Randomly select a mole to show
        if let randomIndex = moles.indices.randomElement() {
            moles[randomIndex].isVisible = true
        }
    }

    func whackMole(at position: Int) {
        guard let index = moles.firstIndex(where: { $0.position == position }) else { return }
        if moles[index].isVisible {
            moles[index].isVisible = false
            score += 1
        }
    }

    func gameCountDown(){
        if timeLeft > 0 {
            timeLeft -= 1
        }
    }
}

struct MoleView: View {
    var isVisible: Bool
    var onMoleTapped: () -> Void

    var body: some View {
        Button(action: onMoleTapped) {
            ZStack {
                Rectangle()
                    .fill(Color.gray)
                    .frame(width: 100, height: 100)
                if isVisible {
                  Image(systemName: "circle.fill")
                        .resizable()
                        .foregroundColor(.red)
                        .frame(width: 80, height: 80)
                        .transition(.scale)
                }
            }
        }
        .disabled(!isVisible) // disable the button if mole is not visible
    }
}

#Preview {
