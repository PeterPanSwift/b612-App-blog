使用 regularExpression 判斷 Swift 字串的語言

判斷中文

func containsChineseCharacters(input: String) -> Bool {
    input.range(of: "\\p{Han}", options: .regularExpression) != nil
}
containsChineseCharacters(input: "蘋果")
containsChineseCharacters(input: "apple")

判斷英文

func containsEnglishCharacters(input: String) -> Bool {
    let pattern = "[a-zA-Z]"
    return input.range(of: pattern, options: .regularExpression) != nil
}
containsEnglishCharacters(input: "apple")
containsEnglishCharacters(input: "蘋果")

判斷日文

func containsJapaneseCharacters(input: String) -> Bool {
    let pattern = "[\\p{Han}\\p{Hiragana}\\p{Katakana}]"
    return input.range(of: pattern, options: .regularExpression) != nil
}
containsJapaneseCharacters(input: "りんご")
containsJapaneseCharacters(input: "apple")
