輸入文字時動畫更新畫面


struct ContentView: View {
    @State private var streetAddress = ""
    @State private var city = ""
    @State private var state = ""
    @State private var zipCode = ""
    @State private var country = ""
    
    let gradientColors = Gradient(colors: [Color.blue.opacity(0.6), Color.purple.opacity(0.6)])
    
    var body: some View {
        ZStack {
            LinearGradient(gradient: gradientColors, startPoint: .topLeading, endPoint: .bottomTrailing)
                .ignoresSafeArea()
            
            ScrollView {
                VStack(spacing: 30) {
                    Text("Address Label")
                        .font(.system(size: 32, weight: .bold, design: .rounded))
                        .foregroundStyle(.white)
                        .padding()
                    
                    // Formatted address display
                    VStack(alignment: .leading, spacing: 4) {
                        Text(streetAddress)
                        Text("\(city), \(state) \(zipCode)")
                        Text(country)
                    }
                    .font(.system(size: 18, weight: .medium, design: .serif))
                    .foregroundStyle(.white)
                    .frame(maxWidth: .infinity, alignment: .leading)
                    .padding()
                    .background(Color.white.opacity(0.2))
                    .clipShape(.rect(cornerRadius: 15))
                    .shadow(radius: 10)
                    
                    // Input form
                    VStack(spacing: 20) {
                        CustomTextField(placeholder: "Street Address", text: $streetAddress)
                        CustomTextField(placeholder: "City", text: $city)
                        CustomTextField(placeholder: "State", text: $state)
                        CustomTextField(placeholder: "Zip Code", text: $zipCode)
                        CustomTextField(placeholder: "Country", text: $country)
                    }
                }
                .padding()
            }
        }
        .animation(.spring(), value: [streetAddress, city, state, zipCode, country])
    }
}

struct CustomTextField: View {
    let placeholder: String
    @Binding var text: String
    
    var body: some View {
        TextField(placeholder, text: $text)
            .padding()
            .background(Color.white)
            .clipShape(.rect(cornerRadius: 10))
            .shadow(color: Color.black.opacity(0.1), radius: 5, x: 0, y: 5)
    }
}
