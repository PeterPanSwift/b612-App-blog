collection view 的 infinite scroll 


@objc func changeBanner() {
        var indexPath: IndexPath
        imageIndex += 1
        print(imageIndex)
        if imageIndex < images.count {
            indexPath = IndexPath(item: imageIndex, section: 0)
            seaCollectionView.selectItem(at: indexPath, animated: true, scrollPosition: .centeredHorizontally)
        }else{
            imageIndex = 0
            indexPath = IndexPath(item: imageIndex, section: 0)
            seaCollectionView.selectItem(at: indexPath, animated: false , scrollPosition: .centeredHorizontally)
            changeBanner()
        }
}
