import SwiftUI

struct ContentView: View {
    // 定義可以吃的食物列表
    private static let foodOptions = [
        "拉麵",
        "壽司",
        "牛排",
        "沙拉",
        "火鍋",
        "披薩",
        "漢堡",
        "炒飯",
        "燒烤",
        "義大利麵"
    ]
    
    // 用於儲存隨機選擇的食物
    @State private var selectedFood: String?
    
    var body: some View {
        VStack(spacing: 20) {
            Text("今天吃什麼呢？")
                .font(.largeTitle)
                .fontWeight(.bold)
            
            // 顯示隨機選擇的食物
            if let food = selectedFood {
                Text(food)
                    .font(.title)
                    .foregroundColor(.blue)
            } else {
                Text("點選按鈕來決定吧！")
                    .font(.title2)
                    .foregroundColor(.gray)
            }
            
            // 隨機選擇食物的按鈕
            Button(action: {
                selectedFood = ContentView.foodOptions.randomElement()
            }) {
                Text("今天吃什麼？")
                    .modifier(CustomButtonStyle())
            }
        }
        .padding()
    }
}

struct CustomButtonStyle: ViewModifier {
    func body(content: Content) -> some View {
        content
            .font(.title)
            .padding()
            .background(Color.pink)
            .foregroundColor(.white)
            .cornerRadius(10)
    }
}

import SwiftUI

struct ContentView: View {
    // 定義可以吃的食物列表，這是一個靜態常量，避免每次實例化 ContentView 時重新創建數組
    private static let foodOptions = [
        "拉麵",
        "壽司",
        "牛排",
        "沙拉",
        "火鍋",
        "披薩",
        "漢堡",
        "炒飯",
        "燒烤",
        "義大利麵"
    ]
    
    // 用於儲存隨機選擇的食物，使用 @State 使得選擇的食物能夠在 UI 中觸發更新
    @State private var selectedFood: String?
    
    var body: some View {
        VStack(spacing: 20) {
            // 標題文字，顯示應用的主要目的
            Text("今天吃什麼呢？")
                .font(.largeTitle) // 設定字體大小為大標題
                .fontWeight(.bold) // 設定字體為粗體
                .onAppear {
                    // 當視圖出現時打印信息，用於調試
                    print("ContentView appeared")
                }
            
            // 根據是否有選擇的食物來顯示不同的文字
            if let food = selectedFood {
                // 如果已選擇食物，顯示選擇的食物名稱
                Text(food)
                    .font(.title) // 設定字體大小為標題
                    .foregroundColor(.blue) // 設定字體顏色為藍色
                    .onAppear {
                        // 當選擇的食物視圖出現時打印選擇的食物，用於調試
                        print("Selected food: \(food)")
                    }
            } else {
                // 如果尚未選擇食物，顯示提示文字
                Text("點選按鈕來決定吧！")
                    .font(.title2) // 設定字體大小為次標題
                    .foregroundColor(.gray) // 設定字體顏色為灰色
                    .onAppear {
                        // 當提示視圖出現時打印信息，用於調試
                        print("No food selected yet")
                    }
            }
            
            // 隨機選擇食物的按鈕
            Button(action: {
                // 點擊按鈕時隨機選擇食物
                selectedFood = ContentView.foodOptions.randomElement()
                // 打印選擇的食物，用於調試
                print("Button tapped, selectedFood: \(String(describing: selectedFood))")
            }) {
                // 按鈕顯示的文字
                Text("今天吃什麼？")
                    .modifier(CustomButtonStyle()) // 使用自定義的按鈕樣式
            }
        }
        .padding() // 添加間距，讓整個 VStack 有足夠的空白空間
    }
}

// 自定義按鈕樣式的 ViewModifier，用於統一按鈕的外觀設計
struct CustomButtonStyle: ViewModifier {
    func body(content: Content) -> some View {
        content
            .font(.title) // 設定字體大小為標題
            .padding() // 添加內部間距
            .background(Color.pink) // 設定背景顏色為粉紅色
            .foregroundColor(.white) // 設定字體顏色為白色
            .cornerRadius(10) // 設定圓角半徑為 10
    }
}
