import UIKit

class IntroductionViewController: UIViewController {
    
    private lazy var dogLabel = {
        let label = UILabel()
        configureAnimalLabel(label: label, emoji: "ğŸ¶")
        return label
    }()
    
    private lazy var catLabel = {
        let label = UILabel()
        configureAnimalLabel(label: label, emoji: "ğŸ±")
        return label
    }()
    
    private lazy var rabbitLabel = {
        let label = UILabel()
        configureAnimalLabel(label: label, emoji: "ğŸ°")
        return label
    }()
    
    private lazy var turtleLabel = {
        let label = UILabel()
        configureAnimalLabel(label: label, emoji: "ğŸ¢")
        return label
    }()
    
    private lazy var titleLabel = {
        let label = UILabel()
        label.text = "Which Animal Are You?"
        label.font = UIFont(name: "Georgia", size: 30)
        label.textAlignment = .center
        return label
    }()
    
    private lazy var startButton = {
        let button = UIButton(type: .system)
        button.setTitle("Begin Personality Quiz", for: .normal)
        button.titleLabel?.font = .systemFont(ofSize: 15)
        button.addTarget(self, action: #selector(showQuestionView), for: .touchUpInside)
        return button
    }()
    
    private lazy var stackView = {
        let stackView = UIStackView()
        stackView.translatesAutoresizingMaskIntoConstraints = false
        stackView.axis = .vertical
        stackView.spacing = 8
        stackView.alignment = .fill
        stackView.distribution = .fill
        return stackView
    }()
    
    
    override func viewDidLoad() {
        super.viewDidLoad()
        // Do any additional setup after loading the view.
        
        view.backgroundColor = .systemBackground
        view.addSubview(dogLabel)
        view.addSubview(catLabel)
        view.addSubview(rabbitLabel)
        view.addSubview(turtleLabel)
        stackView.addArrangedSubview(titleLabel)
        stackView.addArrangedSubview(startButton)
        view.addSubview(stackView)
        configureConstraints()
    }
    
    func configureConstraints() {
        NSLayoutConstraint.activate([            
            dogLabel.leadingAnchor.constraint(equalTo: view.safeAreaLayoutGuide.leadingAnchor, constant: 20),            
            dogLabel.topAnchor.constraint(equalTo: view.safeAreaLayoutGuide.topAnchor),            
            catLabel.leadingAnchor.constraint(greaterThanOrEqualTo: dogLabel.trailingAnchor),            
            catLabel.topAnchor.constraint(equalTo: view.safeAreaLayoutGuide.topAnchor),           
            view.safeAreaLayoutGuide.trailingAnchor.constraint(equalTo: catLabel.trailingAnchor, constant: 20),                        

            rabbitLabel.leadingAnchor.constraint(equalTo: view.safeAreaLayoutGuide.leadingAnchor, constant: 20),            
            rabbitLabel.bottomAnchor.constraint(equalTo: view.safeAreaLayoutGuide.bottomAnchor),            
            turtleLabel.leadingAnchor.constraint(greaterThanOrEqualTo: rabbitLabel.trailingAnchor),            
            turtleLabel.bottomAnchor.constraint(equalTo: view.safeAreaLayoutGuide.bottomAnchor),            
            view.safeAreaLayoutGuide.trailingAnchor.constraint(equalTo: turtleLabel.trailingAnchor, constant: 20),                        

            stackView.centerYAnchor.constraint(equalTo: view.centerYAnchor),           
            stackView.leadingAnchor.constraint(equalTo: view.safeAreaLayoutGuide.leadingAnchor, constant: 8),            
            view.safeAreaLayoutGuide.trailingAnchor.constraint(equalTo: stackView.trailingAnchor, constant: 8)                    
        ])
    }
    
    func configureAnimalLabel(label: UILabel, emoji: String) {
        label.translatesAutoresizingMaskIntoConstraints = false
        label.text = emoji
        label.font = .systemFont(ofSize: 40)
    }

        let controller = storyboard.instantiateViewController(withIdentifier: "\(QuestionViewController.self)")
        let navigationController =  UINavigationController(rootViewController: controller)
        present(navigationController, animated: true)
    }
    
    @IBAction func unwindToQuizIntroduction(segue: UIStoryboardSegue) {
        
    }    
}
