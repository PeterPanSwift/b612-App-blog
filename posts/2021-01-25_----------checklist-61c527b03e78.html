真心不騙的作業項目 checklist

完成作業後，在 App 裡加入真心不騙的作業項目 checklist，描述完成的功能。

以選擇題 App 為例。

#25 選擇題 App
作業目的: 學習變數，for & while 迴圈，if else ，array，類別和物件，亂數，自訂資料型別，viewDidLoad，outlet & action。medium.com

加入 Item.swift，定義作業項目和是否完成

import Foundation
struct Item: Identifiable {
    let id = UUID()
    let name: String
    let done: Bool
}
extension Item {
    static var homework = "選擇題 App"
    static var basicItems: [Item] {
        [
            Item(name: "製作選擇題 App，每題有四個選項。", done: true),
            Item(name: "答對一題加 10 分。", done: true),
            Item(name: "畫面上顯示目前題目是第幾題。", done: false),
            Item(name: "自訂資料型別。", done: false),
            Item(name: "題庫有 n 題，隨機出其中的 10 題，每次玩的時候題目順序都不一樣。( n > 10 )", done: false),
            Item(name: "玩完後可選擇再玩一次，重新開始玩。", done: false),
            Item(name: "同一頁顯示問題跟分數。", done: false),
        ]
    }
    static var bonusItems: [Item] {
        [
           Item(name: "每題四個選項的順序不固定。", done: false),
           Item(name: "分成問題頁跟分數頁，將結果從問題頁傳到分數頁。", done: false),
           Item(name: "題目可從 Airtable 輸入，將輸出的 csv 變成 array。", done: false),
           Item(name: "計時。", done: false),
           Item(name: "儲存玩家分數的歷史記綠和最高分。", done: false),
           Item(name: "將玩家的分數上傳，實現分數排行榜。", done: false),
           Item(name: "搭配線上題庫 API Open Trivia DB。", done: false),
       ]
    }
}

basicItems 描述基本功能，bonusItems 描述進階功能，完成的項目 done 傳入 true，未完成的傳入 false。

加入 CheckList.swift

定義 checklist 的畫面。

import SwiftUI
struct CheckList: View {
    let basicItems = Item.basicItems
    let bonusItems = Item.bonusItems
    
    var body: some View {
        List {
            Section(header: Text("基本功能"), footer: PercentView(items: basicItems)) {
                ForEach(basicItems) { (item) in
                    ItemRow(item: item)
                }
            }
            Section(header: Text("進階功能"), footer: PercentView(items: bonusItems)) {
                ForEach(bonusItems) { (item) in
                    ItemRow(item: item)
                }
            }
        }
        .navigationTitle(Item.homework)
        
    }
}
struct CheckList_Previews: PreviewProvider {
    static var previews: some View {
        CheckList()
            .preferredColorScheme(.dark)
    }
}
struct ItemRow: View {
    let item: Item
    
    var body: some View {
        HStack {
            Text(item.name)
            Spacer()
            Image(systemName: item.done ? "checkmark.rectangle" : "rectangle")
                .foregroundColor(item.done ? .green : .primary)
            
            
            
        }
    }
}
struct PercentView: View {
    let items: [Item]
    
    func getPercentString() -> String {
        let formatter = NumberFormatter()
        formatter.numberStyle = .percent
        let checkCount = items.reduce(0) { (result, item) in
            result + (item.done ? 1 : 0)
        }
        let percent = Double(checkCount) / Double(items.count)
        return formatter.string(from: NSNumber(value: percent))!
    }
    
    var body: some View {
        Text("完成 \(getPercentString())").frame(maxWidth: .infinity, maxHeight: 40).background(Color.green)
    }
}

顯示 checklist 的畫面

加入以下程式，在點選 button 時顯示 checklist 的畫面。

  • SwiftUI App 做法
struct ContentView: View {
    
    @State private var showCheckList = false
    
    var body: some View {
        Button(action: {
            showCheckList = true
        }) {
            Text("show check list")
        }
        .sheet(isPresented: $showCheckList, content: {
            NavigationView {
                CheckList()
            }
        })
    }
}
  • UIKit App 做法
@IBAction func showCheckList(_ sender: Any) {
        let controller = UIHostingController(rootView: CheckList())
        let navController = UINavigationController(rootViewController: controller)
        navController.navigationBar.prefersLargeTitles = true
        present(navController, animated: true, completion: nil)
        
}
