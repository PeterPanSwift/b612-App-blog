import requests
from bs4 import BeautifulSoup
import json

def get_video_info(video_page_url):
    response = requests.get(video_page_url)
    soup = BeautifulSoup(response.text, 'html.parser')

    title = soup.find('title').text  # assuming the title tag contains the video title
    video_url = video_page_url
    image_url = soup.find('img')['src']  # assuming the first img tag contains the video thumbnail
    duration = soup.find('div', {'class': 'video-duration'}).text  # assuming a div with class 'video-duration' contains the video duration
    description = soup.find('div', {'class': 'video-description'}).text  # assuming a div with class 'video-description' contains the video description

    return {
        'title': title,
        'video_url': video_url,
        'image_url': image_url,
        'duration': duration,
        'description': description
    }

def get_all_videos(main_page_url):
    response = requests.get(main_page_url)
    soup = BeautifulSoup(response.text, 'html.parser')

    video_page_urls = [a['href'] for a in soup.find_all('a', {'class': 'video-link'})]  # assuming a tags with class 'video-link' contain the video page urls

    all_videos = [get_video_info(url) for url in video_page_urls]

    return all_videos

all_videos = get_all_videos('https://developer.apple.com/videos/all-videos/')
print(json.dumps(all_videos, indent=4))

AI 產生的 python 爬蟲程式如下。

import requests
from bs4 import BeautifulSoup
import json

def parse_html():
    url = "https://developer.apple.com/videos/all-videos"
    response = requests.get(url)
    soup = BeautifulSoup(response.text, 'html.parser')
     
    video_list = []

    for video in soup.find_all('li', {'class': 'collection-item'}):
        video_dict = {}

        # get the video name
        name_section = video.find('h4', {'class': 'no-margin-bottom video-title'})
        video_dict['name'] = name_section.text if name_section else ""

        # get the video link
        link = video.find('a', {'class': 'video-image-link'})
        video_dict['url'] = "https://developer.apple.com" + link['href'] if link else ""

        # get the video image
        img = link.find('img', {'class': 'video-image medium-scale'})
        video_dict['image'] = img['src'] if img else ""

        # get the video duration
        duration = link.find('span', {'class': 'video-duration'})
        video_dict['duration'] = duration.text if duration else ""

        # get the video description
        description_section = video.find('p', {'class': 'description'})
        video_dict['description'] = description_section.text if description_section else ""

        video_list.append(video_dict)

    # output as json
    with open('video_list.json', 'w', encoding='utf-8') as f:
        json.dump(video_list, f, ensure_ascii=False, indent=4)

if __name__ == "__main__":
    parse_html()

import Foundation

struct Video: Codable, Identifiable {
    let id = UUID()
    let name: String
    let url: String
    let image: String
    let duration: String
    let description: String
}
  • ContentView.swift
import SwiftUI

struct ContentView: View {
    @State var videos: [Video] = []

    var body: some View {
        NavigationView {
            List(videos) { video in
                VStack(alignment: .leading) {
                    Link(destination: URL(string: video.url)!, label: {
                        VStack(alignment: .leading) {
                            AsyncImage(url: URL(string: video.image)) { image in
                                image.resizable()
                            } placeholder: {
                                ProgressView()
                            }
                            .aspectRatio(contentMode: .fill)
                            .frame(width: 100, height: 100)
                            .clipped()
                            
                            Text(video.name)
                                .font(.headline)
                            Text(video.duration)
                                .font(.subheadline)
                            Text(video.description)
                                .font(.caption)
                        }
                    })
                }
            }
            .onAppear(perform: loadData)
            .navigationTitle("WWDC Videos")
        }
    }

    func loadData() {
        guard let url = URL(string: "https://raw.githubusercontent.com/PeterPanSwift/ParseWWDC/main/video_list.json") else {
            return
        }

        URLSession.shared.dataTask(with: url) { (data, response, error) in
            if let data = data {
                if let decodedResponse = try? JSONDecoder().decode([Video].self, from: data) {
                    DispatchQueue.main.async {
                        self.videos = decodedResponse
                    }
                    return
                }
            }
            print("Fetch failed: \(error?.localizedDescription ?? "Unknown error")")
        }.resume()
    }
}
