nonisolated
func phaseAnimator<Phase>(
    _ phases: some Sequence,
    @ViewBuilder content: @escaping (PlaceholderContentView<Self>, Phase) -> some View,
    animation: @escaping (Phase) -> Animation? = { _ in .default }
) -> some View where Phase : Equatable
phaseAnimator(_:content:animation:) | Apple Developer Documentation
Animates effects that you apply to a view over a sequence of phases that change continuously.developer.apple.com


import SwiftUI

struct ContentView: View {
    var body: some View {
        Image(systemName: "swift")
            .resizable()
            .scaledToFit()
            .foregroundStyle(.orange)
            .phaseAnimator([0, 1]) { content, phase in
                content
                    .opacity(phase)
            }
    }
}
  • step 1: 一開始先在 0
  • step 2: 動畫到 1，
  • step 3: 動畫到 0。
  • step 4: 重覆 step2 & 3。

調整動畫的時間和速度。


import SwiftUI

struct ContentView: View {
    var body: some View {
        Image(systemName: "swift")

            .scaledToFit()
            .foregroundStyle(.orange)
            .phaseAnimator([0, 1]) { content, phase in
                content
                    .opacity(phase)
            } animation: { phase in
                    .easeInOut(duration: 5)
            }

    }
}

import SwiftUI

struct ContentView: View {

            .foregroundStyle(.orange)
            .phaseAnimator([0, 1]) { content, phase in
                content
                    .opacity(phase)
            } animation: { phase in
                phase == 1 ? .easeInOut(duration: 5) : .linear(duration: 0.5)
            }

    }
}

事件觸發的一次性動畫

import SwiftUI

struct ContentView: View {
    @State private var tapCount = 0
    
    var body: some View {
        Image(systemName: "swift")
            .resizable()
            .scaledToFit()
            .phaseAnimator([Color.orange, Color.blue, Color.green], trigger: tapCount, content: { content, phase in
                content
                    .foregroundStyle(phase)
            })
            .onTapGesture {
                tapCount += 1
            }
    }
}
