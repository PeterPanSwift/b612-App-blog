用 swift 寫計算梯形面積的 function

func calculateTrapezoidArea(topLength: Double, bottomLength: Double, height: Double) -> Double {
    return (topLength + bottomLength) * height / 2
}

let area = calculateTrapezoidArea(topLength: 2.0, bottomLength: 4.0, height: 3.0)
print(area) // 9.0

用 swift 寫計算 BMR 的 function

func calculateBMR(weight: Double, height: Double, age: Int, gender: String) -> Double {
    let bmr: Double
    if gender == "male" {
        bmr = 66 + (6.2 * weight) + (12.7 * height) - (6.76 * Double(age))
    } else {
        bmr = 655 + (4.35 * weight) + (4.7 * height) - (4.7 * Double(age))
    }
    return bmr
}

let bmr = calculateBMR(weight: 70.0, height: 175.0, age: 25, gender: "male")
print(bmr) // 1729.8
