SwiftUI 產生 Binding 的各種方法

方法 1: 在 @State，@Binding，@StateObject，@ObservedObject，@EnvironmentObject 的 property 加 $

範例圖片

  • 在 @State property 加 $
struct  Book {   
    var name: String
    var isFinish: Bool
}

struct ContentView: View {
    @State var book = Book(name: "本店招牌菜", isFinish: true)
    
    var body: some View {
        
        VStack {
            Image(book.name)
            Text(book.name)
                .font(.largeTitle)
            Toggle("已讀", isOn: $book.isFinish)
        }
        .padding()
    }
}
  • 在 @Binding property 加 $
struct  Book {   
    var name: String
    var isFinish: Bool
}

struct ContentView: View {
    @State var book = Book(name: "本店招牌菜", isFinish: true)
    
    var body: some View {
        
        BookView(book: $book)
        
    }
}

struct BookView: View {
    
    @Binding var book: Book
    
    var body: some View {
        VStack {
            Image(book.name)
            Text(book.name)
                .font(.largeTitle)
            Toggle("已讀", isOn: $book.isFinish)
        }
        .padding()
    }
}
  • 在 @StateObject 或 @ObservedObject 的 property 加 $
class Book: ObservableObject {
    init(name: String, isFinish: Bool) {
        self.name = name
        self.isFinish = isFinish
    }
    
    var name: String
    var isFinish: Bool
}

struct ContentView: View {
    @StateObject var book = Book(name: "本店招牌菜", isFinish: true)
   
    var body: some View {
        
        VStack {
            Image(book.name)
            Text(book.name)
                .font(.largeTitle)
            Toggle("已讀", isOn: $book.isFinish)
        }
        .padding()
    }
}
  • 在 @EnvironmentObject 的 property 加 $
class Book: ObservableObject {
    init(name: String, isFinish: Bool) {
        self.name = name
        self.isFinish = isFinish
    }
    
    var name: String
    var isFinish: Bool
}

struct ContentView: View {
    @EnvironmentObject var book: Book
    
    var body: some View {
        
        VStack {
            Image(book.name)
            Text(book.name)
                .font(.largeTitle)
            Toggle("已讀", isOn: $book.isFinish)
        }
        .padding()
    }
}

struct ContentView_Previews: PreviewProvider {
    static var previews: some View {
        ContentView()
            .preferredColorScheme(.dark)
            .environmentObject(Book(name: "本店招牌菜", isFinish: true))
    }
}

方法 2: 利用 constant 產生 Binding

通常用在 preview，只是為了讓 preview 有個能顯示的假資料，App 執行時資料會來自網路或前一頁。

struct ContentView: View {
    @Binding var book: Book
    
    var body: some View {
        
        VStack {
            Image(book.name)
            Text(book.name)
                .font(.largeTitle)
            Toggle("已讀", isOn: $book.isFinish)
        }
        .padding()
    }
}

struct ContentView_Previews: PreviewProvider {
    static var previews: some View {
        ContentView(book: .constant(Book(name: "本店招牌菜", isFinish: true)))
            .preferredColorScheme(.dark)
    }
}

不過 function constant 產生的資料是不能改變的，因此下圖的開關將一直開著，無法關掉!
