解答

  • Swift 版。
import Foundation

print("請輸入 GPU 的選擇: 1 - 60 核心，2 - 76 核心 :")
let gpu = readLine()
print("請輸入記憶體的選擇: 1 - 64 GB，2 - 128 GB，3 - 192 GB :")
let memory = readLine()
print("請輸入儲存裝置的選擇: 1 - 1TB，2 - 2TB，3 - 4TB，4 - 8TB :")
let ssd = readLine()
print("請輸入巧控滑鼠或巧控板的選擇: 1 - 巧控滑鼠，2 - 巧控板，3 - 巧控滑鼠 + 巧控板 :")
let control = readLine()

var price = 244900
if gpu == "2" {
    price += 30000
}
if memory == "2" {
    price += 24000
} else if memory == "3" {
    price += 48000
}
if ssd == "2" {
    price += 12000
} else if ssd == "3" {
    price += 30000
} else if ssd == "4" {
    price += 66000
}
if control == "2" {
    price += 1500
} else if control == "3" {
    price += 4290
}

print("只要 $\(price) 就能買到強大的 Mac Pro")
  • Python 版。
gpu = input("請輸入 GPU 的選擇: 1 - 60 核心，2 - 76 核心 :")
