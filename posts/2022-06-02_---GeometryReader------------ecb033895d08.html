利用 GeometryReader 實現元件一樣大的排版

元件一樣大 & 固定高度

struct ContentView: View {
    var body: some View {
        HStack {
            ForEach(1..<3) { index in
                GeometryReader { proxy in
                    Image("peter\(index)")
                        .resizable()
                        .scaledToFill()
                        .frame(width: proxy.size.width)
                        .clipped()
                }
            }
        }
        .frame(height: 250)
        .padding()
    }
}

GeometryReader 將佔滿可用的空間，因此 HStack 裡用 ForEach 產生的多個 GeometryReader 將平分可用空間。在此我們將圖片的寬度設為 proxy.size.width，讓圖片的寬度等於 GeometryReader 的寬度。

元件一樣大 & 固定比例

struct ContentView: View {
    var body: some View {
        HStack {
            ForEach(1..<3) { index in
                GeometryReader { proxy in
                    Image("peter\(index)")
                        .resizable()
                        .scaledToFill()
                        .frame(width: proxy.size.width)
                        .clipped()
                }
                .aspectRatio(1, contentMode: .fit)
            }
        }
        .padding()
    }
}

從 GeometryReader 呼叫 .aspectRatio(1, contentMode: .fit)，讓 GeometryReader 成為正方形。

包含圖文 & 固定比例的圖片

struct ContentView: View {
    var body: some View {
        HStack {
            ForEach(1..<3) { index in
                VStack {
                    GeometryReader { proxy in
                        VStack {
                            Image("peter\(index)")
                                .resizable()
                                .scaledToFill()
                                .frame(width: proxy.size.width)
                                .clipped()
                        }
                    }
                    .aspectRatio(1, contentMode: .fit)
                    
                    Text("peter\(index)")
                        .font(.largeTitle)
                }
            }
        }
        .padding()
    }
}

struct ContentView: View {
    let letters = ["a", "p", "p", "l", "e"]
var body: some View {
        HStack {
            ForEach(letters, id: \.self) { letter in
                GeometryReader { proxy in
                    Text(letter)
                        .font(.largeTitle)
                        .frame(maxWidth: .infinity, maxHeight: .infinity)
                }
                .aspectRatio(1, contentMode: .fit)
                .background(.yellow)
            }
        }
        .padding()
    }
}
