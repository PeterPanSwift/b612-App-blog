利用 didSet 設定 outlet 的內容


class GridViewCell: UICollectionViewCell {
    
    @IBOutlet var imageView: UIImageView!
       
    var thumbnailImage: UIImage! {
        didSet {
            imageView.image = thumbnailImage
        }
    }
