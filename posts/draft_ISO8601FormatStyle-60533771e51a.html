import Foundation


try! Date.ISO8601FormatStyle().parse("2018-12-01T10:00:05.000Z")



struct TimeData: Codable {
    let time: Date
}

let data = """
{
    "time": "2018-12-01T10:00:05.000Z"
}
""".data(using: .utf8)!
let decoder = JSONDecoder()
decoder.dateDecodingStrategy = .iso8601
do {
    let timeData = try decoder.decode(TimeData.self, from: data)
} catch {
    print(error)
}
