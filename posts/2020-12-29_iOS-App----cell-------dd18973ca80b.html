SwiftUI 的 List，搭配定義 cell 畫面的型別

cell 透過 SongRow 定義。

SongList

struct SongList: View {
    let songs = Song.data

    var body: some View {
        List(songs, id: \.name) { song in
            SongRow(song: song)
        }
    }
}

SongRow

struct SongRow: View {
    let song: Song
    
    var body: some View {
        HStack(spacing: 10) {
            Image(song.albumImage)
                .resizable()
                .scaledToFill()
                .frame(width: 40, height: 40)
                .clipShape(.rect(cornerRadius: 6))
            VStack(alignment: .leading, spacing: 4) {
                Text(song.name)
                Text(song.singer)
                    .font(.caption)
                    .foregroundStyle(.secondary)
                    
            }
            Spacer()
            Button(action: {
                print("tap")
            }, label: {
                Image(systemName: "plus")
                    .foregroundStyle(.red)
                
            })
            .buttonStyle(.plain)
        }
    }
}

SwiftUI 的 Grid，搭配定義 cell 畫面的型別

cell 透過 SongRow 定義。

struct SongList: View {
    let songs = Song.data
    
    var body: some View {
        ScrollView(.vertical) {
            let columns = [GridItem()]
            LazyVGrid(columns: columns) {
                ForEach(songs, id: \.name) { song in
                    SongRow(song: song)
                        .padding()
                }
            }
            
        }
    }
}

UIKit 的 table & collection view，從 storyboard 設計 cell

自訂 cell 類別

class SongTableViewCell: UITableViewCell {
    @IBOutlet weak var singerLabel: UILabel!
    @IBOutlet weak var nameLabel: UILabel!
    @IBOutlet weak var albumImageView: UIImageView!
    
    func update(with song: Song) {
        nameLabel.text = song.name
        singerLabel.text = song.singer
        albumImageView.image = UIImage(named: song.albumImage)
    }
}

指定 cell id。

table view 利用 dequeueReusableCell(withIdentifier:for:) 產生 storyboard 設計的 cell，傳入在 storyboard cell 設定的 id。

class SongTableViewController: UITableViewController {
    
    let songs = Song.data

    override func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        return songs.count
    }

    override func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "\(SongTableViewCell.self)", for: indexPath) as! SongTableViewCell
        let song = songs[indexPath.row]
        cell.update(with: song)
        return cell
    }
}

UIKit 的 table & collection view，從 xib 設計 cell

產生 cell 類別時勾選 Also Create XIB file。

在 xib 設計 cell。

定義 cell 類別。

class SongTableViewCell: UITableViewCell {
    @IBOutlet weak var singerLabel: UILabel!
    @IBOutlet weak var nameLabel: UILabel!
    @IBOutlet weak var albumImageView: UIImageView!
    
    func update(with song: Song) {
        nameLabel.text = song.name
        singerLabel.text = song.singer
        albumImageView.image = UIImage(named: song.albumImage)
    }
}

table view 利用 dequeueReusableCell(withIdentifier:for:) 產生 xib 設計的 cell，在 viewDidLoad 時 table view 要先呼叫 register(_:forCellReuseIdentifier:) 註冊 cell 對應的 xib & id。

class SongTableViewController: UITableViewController {
    
    let songs = Song.data
    
    override func viewDidLoad() {
        super.viewDidLoad()
        let name = "\(SongTableViewCell.self)"
        tableView.register(UINib(nibName: name, bundle: nil), forCellReuseIdentifier: name)
    }

    override func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        return songs.count
    }

    override func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "\(SongTableViewCell.self)", for: indexPath) as! SongTableViewCell
        let song = songs[indexPath.row]
        cell.update(with: song)
        return cell
    }
}

UIKit 的 table & collection view，從程式設計 cell

定義 table view cell 類別，在 init(style:reuseIdentifier:) 設定 cell 畫面，在此我們將設定畫面的相關程式定義在 function setupUI，然後在 init 裡呼叫 setupUI。(ps: 若是 collection view cell，則要定義 init(frame:) )

class SongTableViewCell: UITableViewCell {
    
    let singerLabel = UILabel()
    let nameLabel = UILabel()
    let albumImageView = UIImageView()
    
    
    override init(style: UITableViewCell.CellStyle, reuseIdentifier: String?) {
        super.init(style: style, reuseIdentifier: reuseIdentifier)
        setupUI()
    }
    
    required init?(coder: NSCoder) {
        fatalError("init(coder:) has not been implemented")
    }
    
    private func setupUI() {
        let addButton = UIButton()
        addButton.setImage(UIImage(systemName: "plus"), for: .normal)
        let nameStackView = UIStackView(arrangedSubviews: [
            nameLabel,
            singerLabel
        ])
        nameStackView.alignment = .leading
        nameStackView.axis = .vertical
        let songStackView = UIStackView(arrangedSubviews: [
            albumImageView,
            nameStackView,
            addButton
        ])
        songStackView.spacing = 10
        songStackView.axis = .horizontal
        
        albumImageView.translatesAutoresizingMaskIntoConstraints = false
        songStackView.translatesAutoresizingMaskIntoConstraints = false
        
        addSubview(songStackView)
        
        let padding: CGFloat = 10
        NSLayoutConstraint.activate([          albumImageView.widthAnchor.constraint(equalToConstant: 40),
            albumImageView.widthAnchor.constraint(equalTo: albumImageView.heightAnchor),
            songStackView.leadingAnchor.constraint(equalTo: leadingAnchor, constant: padding),
            songStackView.trailingAnchor.constraint(equalTo: trailingAnchor, constant: -padding),
            songStackView.topAnchor.constraint(equalTo: topAnchor, constant: padding),
            songStackView.bottomAnchor.constraint(equalTo: bottomAnchor, constant: -padding),
        ])
    }
    
    func update(with song: Song) {
        nameLabel.text = song.name
        singerLabel.text = song.singer
        albumImageView.image = UIImage(named: song.albumImage)
    }
}

在 controller 的 viewDidLoad 裡，table view 要先呼叫 register(_:forCellReuseIdentifier:) 註冊 cell 對應的類別 & id。

class SongTableViewController: UITableViewController {
    
    let songs = Song.data
    
    override func viewDidLoad() {
        super.viewDidLoad()
        let name = SongTableViewCell.self
        tableView.register(name, forCellReuseIdentifier: "\(name)")
    }
