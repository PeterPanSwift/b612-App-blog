Swift 3 的 selector 語法!

以 #selector 搭配 Xcode 幫我們完成的 function 名稱。

func playMusic(name:String) {
}
override func viewDidLoad() {
    super.viewDidLoad()
    let button = UIButton()
    button.addTarget(self, action: #selector(ViewController.playMusic(name:)), for: .touchUpInside)
}
