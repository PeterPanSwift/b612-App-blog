在 Swift 程式加上 /// 開頭的 documentation comment 可以生成幫助開發的 API 文件，用以下 Apple 舉的 eat function 為例。

struct Food {
    let name: String
    let energy: Int
}

struct Sloth {
    var energyLevel = 5
    /// Eat the provided specialty sloth food.
    ///
    /// Sloths love to eat while they move very slowly through their rainforest
    /// habitats. They're especially happy to consume leaves and twigs, which they
    /// digest over long periods of time, mostly while they sleep.
    ///
    /// When they eat food, a sloth's `energyLevel` increases by the food's `energy`.
    ///
    /// - Parameters:
    ///   - food: The food for the sloth to eat.
    ///   - quantity: The quantity of the food for the sloth to eat.
    ///
    /// - Returns: The sloth's energy level after eating.
    ///
    /// - Throws: `SlothError.tooMuchFood` if the quantity is more than 100.
    mutating public func eat(_ food: Food, quantity: Int) throws -> Int {
        energyLevel += food.energy * quantity
        return energyLevel
    }
}

import SwiftUI

/// 主要的 `ScrumStore` 類別，用於管理日常 Scrum 數據。
///
/// 這個類別遵循 `ObservableObject` 協議，允許 SwiftUI 視圖根據數據的變化自動更新。
/// 它負責保存和加載日常 Scrum 會議的數據。
@MainActor
class ScrumStore: ObservableObject {
    /// 發布 scrums 數據的陣列，用於追蹤和顯示所有日常 Scrum 會議。
    @Published var scrums: [DailyScrum] = []
    
    /// 私有靜態方法，用於計算 scrums 數據文件的存儲 URL。
    ///
    /// - Throws: 可能拋出文件管理錯誤。
    /// - Returns: 用於存儲 scrums 數據的文件 URL。
    private static func fileURL() throws -> URL {
        try FileManager.default.url(for: .documentDirectory,
                                    in: .userDomainMask,
                                    appropriateFor: nil,
                                    create: false)
        .appendingPathComponent("scrums.data")
    }
    
    /// 異步方法，用於加載保存在本地的 scrums 數據。
    ///
    /// - Throws: 可能拋出數據加載或解碼錯誤。
    func load() async throws {
        let task = Task<[DailyScrum], Error> {
            // 計算文件 URL 並從中加載數據。
            let fileURL = try Self.fileURL()
            guard let data = try? Data(contentsOf: fileURL) else {
                return []
            }
            // 解碼數據以獲取日常 Scrum 陣列。
            let dailyScrums = try JSONDecoder().decode([DailyScrum].self, from: data)
            return dailyScrums
        }
        // 獲取加載的數據並更新 scrums 屬性。
        let scrums = try await task.value
        self.scrums = scrums
    }
    
    /// 異步方法，用於將 scrums 數據保存到本地文件。
    ///
    /// - Parameter scrums: 要保存的日常 Scrum 陣列。
    /// - Throws: 可能拋出數據編碼或文件寫入錯誤。
    func save(scrums: [DailyScrum]) async throws {
        let task = Task {
            // 編碼 scrums 數據為 JSON。
            let data = try JSONEncoder().encode(scrums)
            // 計算文件 URL 並將數據寫入文件。
            let outfile = try Self.fileURL()
            try data.write(to: outfile)
        }
        // 等待保存任務完成。
        _ = try await task.value
    }
}
