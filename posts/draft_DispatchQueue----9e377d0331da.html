DispatchQueue 問題


let serialQueue = DispatchQueue(label: "swiftlee.serial.queue")
let group = DispatchGroup()
func test() {
    group.enter()
    serialQueue.async {
        print("1 start ")
        
        serialQueue.async {
            print("2 start ")
            
            serialQueue.async {
                print("3 start ")
                print("3 end ")
                group.leave()
            }
            
            print("2 end ")
}
        print("1 end ")
}
    group.wait()
}
for i in 1...3 {
    print("for i", i)
    test()
}

結果

for 卡住 main thread 

for i 1
1 start 
1 end 
2 start 
2 end 
3 start 
3 end 
for i 2

3 end 
for i 3

2 end

解法1

DispatchQueue.global().async {
    for i in 1...3 {
        print("for i", i)
        test()
    }
}

解法 2

for i in 1...3 {
    print("for i", i)
    DispatchQueue.global().async {
        test()
        
    }
}
