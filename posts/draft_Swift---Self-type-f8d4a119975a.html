Swift 的 Self type


呼叫型別的 property or method ，不用再打型別的名字。

class Baby {
static var maxAge = 100
static func test() {

}

func test() {
print(Self.maxAge)
Self.test()
}
}


a & b 的型別是 Baby

import UIKit

var str = "Hello, playground"

class Baby {
var name: String
required init(name: String) {
self.name = name
}

static func test2() -> Baby {
return self.init(name: "peter")
}

func test() -> Baby {

return self
}

}

class SuperBaby: Baby {

}
var cuteBaby = SuperBaby(name: "hook")
var a = cuteBaby.test()
var b = SuperBaby.test2()
