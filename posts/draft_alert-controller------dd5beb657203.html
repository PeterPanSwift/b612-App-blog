let controller = UIAlertController(title: "", message: "", preferredStyle: .actionSheet)
let vc = UIViewController()
vc.preferredContentSize = CGSize(width: view.bounds.width,height: 300)
dataPicker = UIPickerView(frame: CGRect(x: 0, y: 0, width: view.bounds.width, height: 300))
dataPicker?.delegate = self
dataPicker?.dataSource = self
vc.view.addSubview(dataPicker!)

controller.setValue(vc, forKey: "contentViewController")
