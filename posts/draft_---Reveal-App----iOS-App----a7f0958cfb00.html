設計 App 畫面

override func viewDidLoad() {
        super.viewDidLoad()
        // Do any additional setup after loading the view.
        
        let greenView = UIView(frame: CGRect(x: 0, y: 0, width: 200, height: 200))
        greenView.backgroundColor = UIColor.green
        view.addSubview(greenView)
        
        var path = UIBezierPath()
        path.move(to: CGPoint(x: 0, y: 50))
        path.addLine(to: CGPoint(x: 100, y: 0))
        path.addLine(to: CGPoint(x: 100, y: 100))
        path.close()
        let triangleLayer1 = CAShapeLayer()
        triangleLayer1.path = path.cgPath
        triangleLayer1.fillColor = CGColor(srgbRed: 0, green: 0, blue: 1, alpha: 1)
        view.layer.addSublayer(triangleLayer1)
        
        path = UIBezierPath()
        path.move(to: CGPoint(x: 50, y: 150))
        path.addLine(to: CGPoint(x: 100, y: 0))
        path.addLine(to: CGPoint(x: 200, y: 200))
        path.close()
        let triangleLayer2 = CAShapeLayer()
        triangleLayer2.path = path.cgPath
        triangleLayer2.fillColor = UIColor.clear.cgColor
        triangleLayer2.lineWidth = 10
        triangleLayer2.strokeColor = CGColor(srgbRed: 1, green: 0, blue: 0, alpha: 1)
        view.layer.addSublayer(triangleLayer2)
        
        
        let blackView = UIView(frame: CGRect(x: 100, y: 100, width: 80, height: 80))
        blackView.backgroundColor = UIColor.black
        view.addSubview(blackView)
        
}
