import UIKit
let view = UIView(frame: CGRect(x: 0, y: 0, width: 200, height: 200))
view.backgroundColor = UIColor.blue
let path = UIBezierPath(roundedRect: CGRect(x: 0, y: 0, width: 200, height: 200), cornerRadius: 30)
let eyePath = UIBezierPath(ovalIn: CGRect(x: 100, y: 100, width: 30, height: 30))
path.append(eyePath)
let layer = CAShapeLayer()
layer.path = path.cgPath
layer.fillRule = .evenOdd
view.layer.mask = layer
