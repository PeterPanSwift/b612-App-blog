BorderlessButtonStyle

讓 List 裡的多個 button 可以分開點選

BorderlessButtonStyle


  • 讓 List 裡的多個 button 可以分開點選 
  • 多個 NavigationLink
struct ContentView: View {
    let movies = [Movie(name: "蜘蛛人", actor: "Tom"), Movie(name: "小丑", actor: "Joaquin")]
    @State private var selectMovie: String?
    @State private var selectActor: String?
    
    var body: some View {
        NavigationView {
            List(movies, id: \.name) { (movie) in
                HStack {
                    NavigationLink(destination: MovieDetail(name: movie.name), tag: movie.name, selection: self.$selectMovie) {
                        EmptyView()
                    }
                    .frame(width: 0, height: 0)
                    
                    NavigationLink(destination: ActorView(name: movie.actor), tag: movie.actor, selection: self.$selectActor) {
                        EmptyView()
                    }
                    .frame(width: 0, height: 0)
                    
                    Button(action: {
                        self.selectMovie = movie.name
                    }) {
                        Text(movie.name)
                    }
                    .buttonStyle(BorderlessButtonStyle())
                    
                    
                    Button(action: {
                        self.selectActor = movie.actor
                    }) {
                        Text(movie.actor)
                    }
                    .buttonStyle(BorderlessButtonStyle())
                    
                }
            }
        }
    }
    
}
