SwiftUI TextField 邊框的設計


邊框顏色太淺。

struct ContentView: View {
    @State private var name = ""
    
    var body: some View {
        TextField("name", text: $name)
            .textFieldStyle(.roundedBorder)
            .padding()
    }
}

邊框高度不夠高。

struct ContentView: View {
    @State private var name = ""
    
    var body: some View {
        TextField("name", text: $name)
            .overlay(content: {
                RoundedRectangle(cornerRadius: 4)
                    .stroke(.gray, lineWidth: 2)
            })
            .padding()
    }
}
