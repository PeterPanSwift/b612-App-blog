import FirebaseFirestoreSwift

struct Singer: Codable, Identifiable {
    @DocumentID var id: String?
    let name: String
    //存專輯名字
    let albums: [String]
}


import SwiftUI
import FirebaseFirestore
import FirebaseFirestoreSwift

struct ContentView: View {
    var body: some View {
        VStack {
            Button("check Taylor Swift") {
                checkTaylorSwift()
            }
            
        }
        .padding()
        
    }
    
    func checkTaylorSwift() {
        let db = Firestore.firestore()
        db.collection("singers").whereField("name", isEqualTo: "Taylor Swift").addSnapshotListener { snapshot, error in
            guard let snapshot else { return }
            snapshot.documentChanges.forEach { documentChange in
                guard let singer = try? documentChange.document.data(as: Singer.self) else { return }
                switch documentChange.type {
                case .added:
                    print("add", singer)
                case .modified:
                    print("modified", singer)
                case .removed:
                    print("removed", singer)
                }
            }
        }
    }
}

import SwiftUI
import FirebaseFirestore
import FirebaseFirestoreSwift

struct ContentView: View {
    var body: some View {
        VStack {
             
            Button("check Taylor Swift") {
                checkTaylorSwift()
            }
            
        }
        .padding()
        
    }
    
    func checkTaylorSwift() {
        let db = Firestore.firestore()
        db.collection("singers").whereField("name", isEqualTo: "Taylor Swift").addSnapshotListener { snapshot, error in
            guard let snapshot else { return }
            snapshot.documentChanges.forEach { documentChange in
                guard let singer = try? documentChange.document.data(as: Singer.self) else { return }
                for albumID in singer.albums {
                    db.collection("albums").document(albumID).getDocument { snapshot, error in
                        guard let snapshot else { return }
                        guard let album = try? snapshot.data(as: Album.self) else {
                            return
                        }
                        print(album)
                    }
                }
               
                switch documentChange.type {
                case .added:
                    print("add", singer)
                case .modified:
                    print("modified", singer)
                case .removed:
                    print("removed", singer)
                }
            }
        }
    }
}
