請 AI 閱讀 API 文件產生 Swift 程式

以下連結是 API 文件，請幫我寫出串接 Create User 的 Swift 程式 https://favqs.com/api

import Foundation

struct User: Codable {
    let login: String
    let email: String
    let password: String
}

let user = User(login: "user_login_here", email: "user_email_here", password: "user_password_here")

let url = URL(string: "https://favqs.com/api/users")!
var request = URLRequest(url: url)
request.httpMethod = "POST"
request.setValue("application/json", forHTTPHeaderField: "Content-Type")
request.setValue("Token token=\"YOUR_APP_TOKEN\"", forHTTPHeaderField: "Authorization")

let encoder = JSONEncoder()
let jsonData = try encoder.encode(user)
request.httpBody = jsonData

let task = URLSession.shared.dataTask(with: request) { data, response, error in
    if let error = error {
        print("Error: \(error)")
    } else if let data = data {
        let decoder = JSONDecoder()
        do {
            let response = try decoder.decode(User.self, from: data)
            print("User created with login: \(response.login)")
        } catch {
            print("Error: \(error)")
        }
    }
}

task.resume()

以下連結是 API 文件，請用 Swift 寫出讀取 Object 的程式 https://metmuseum.github.io

import Foundation

// 定義物件和物件列表的結構
struct Object: Decodable {
    let objectID: Int
    let isHighlight: Bool
    let accessionNumber: String
    let accessionYear: String
    let isPublicDomain: Bool
    let primaryImage: String
    let primaryImageSmall: String
    let additionalImages: [String]
    let constituents: [Constituent]
    let department: String
    let objectName: String
    let title: String
    let culture: String
}

struct Constituent: Decodable {
    let constituentID: Int
    let role: String
    let name: String
    let constituentULAN_URL: String
    let constituentWikidata_URL: String
    let gender: String
}

struct ObjectList: Decodable {
    let total: Int
    let objectIDs: [Int]
}

// 從 API 取得物件列表
let url = URL(string: "https://collectionapi.metmuseum.org/public/collection/v1/objects")!
let task = URLSession.shared.dataTask(with: url) { (data, response, error) in
    if let data = data {
        let decoder = JSONDecoder()
        if let objectList = try? decoder.decode(ObjectList.self, from: data) {
            // 取得第一個物件的詳細資訊
            let objectID = objectList.objectIDs.first!
            let objectURL = URL(string: "https://collectionapi.metmuseum.org/public/collection/v1/objects/\(objectID)")!
            let objectTask = URLSession.shared.dataTask(with: objectURL) { (objectData, objectResponse, objectError) in
                if let objectData = objectData {
                    if let object = try? decoder.decode(Object.self, from: objectData) {
                        // 這裡可以處理物件資訊
                        print(object)
                    }
                }
            }
            objectTask.resume()
        }
    }
}
task.resume()
