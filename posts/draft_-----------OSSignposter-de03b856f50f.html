import UIKit
import os.signpost

class ViewController: UIViewController {
    let signposter = OSSignposter()
    lazy var signpostID = signposter.makeSignpostID()
    var intervalState: OSSignpostIntervalState?
    
    override func viewDidLoad() {
        super.viewDidLoad()
        intervalState = signposter.beginInterval("processRequest", id: signpostID)
        DispatchQueue.main.asyncAfter(deadline: .now() + 5) { [self] in
            if let intervalState  {
                signposter.endInterval("processRequest", intervalState)
            }
        }
    }

    override func viewDidAppear(_ animated: Bool) {
        super.viewDidAppear(animated)
        
    }
}
