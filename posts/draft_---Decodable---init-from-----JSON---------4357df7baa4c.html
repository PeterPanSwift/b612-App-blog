透過 JSONDecoder，我們可以方便地將 JSON 資料變成遵從 protocol Decodable 的自訂型別，而且大部分的時候我們只要宣告型別裡的 property，不用自己撰寫解析 JSON 的 init(from…medium.com



struct User: Decodable {
    let id: Int
    let age: Int
    let comment: String
    
    enum CodingKeys: CodingKey {
        case id, age, comment
    }
    
    init(from decoder: Decoder) throws {
        let container = try decoder.container(keyedBy: CodingKeys.self)
        comment = try container.decode(String.self, forKey: .comment)
        let idString = try container.decode(String.self, forKey: .id)
        guard let idNumber = Int(idString) else {
            let context = DecodingError.Context(
                codingPath: [CodingKeys.id],
                debugDescription: "id error")
            throw DecodingError.dataCorrupted(context)
        }
        id = idNumber
        let ageString = try container.decode(String.self, forKey: .age)
        guard let ageNumber = Int(ageString) else {
            let context = DecodingError.Context(
                codingPath: [CodingKeys.age],
                debugDescription: "age error")
            throw DecodingError.dataCorrupted(context)
        }
        age = ageNumber
    }
}
let url = URL(string: "https://sheetdb.io/api/v1/58f61be4dda40")!
URLSession.shared.dataTask(with: url) { (data, response, error) in
    try! JSONDecoder().decode([User].self, from: data!)
    if let data = data, let users = try? JSONDecoder().decode([User].self, from: data) {
