Flutter ListView

用 Flutter 的 ListView 顯示 IMDb 2024 電影排行榜前十名
import 'package:flutter/material.dart';

// 定義電影資料模型
class Movie {
  final int rank;
  final String title;
  final double rating;
  final String description;

  Movie({
    required this.rank,
    required this.title,
    required this.rating,
    required this.description,
  });
}

class MovieListPage extends StatelessWidget {
  // IMDb 2024 電影排行榜前十名（示範用資料，請勿認真哦～）
  final List<Movie> movies = [
    Movie(rank: 1, title: "死侍與金剛狼", rating: 8.0, description: "漫威宇宙大碰撞，死侍與金剛狼攜手創造歷史！"),
    Movie(rank: 2, title: "沙丘2", rating: 8.5, description: "保羅亞崔迪再次上路，沙丘星球的冒險持續延伸。"),
    Movie(rank: 3, title: "疯狂的麦克斯：狂暴女神", rating: 7.5, description: "末日狂飆中展現女力，狂暴女神霸氣登場！"),
    Movie(rank: 4, title: "小丑2：雙重妄想", rating: 7.2, description: "小丑的心路歷程更進一步，雙重人格大揭秘。"),
    Movie(rank: 5, title: "異形：奪命艦", rating: 7.3, description: "科幻與恐怖交織，異形來襲讓你大呼過癮。"),
    Movie(rank: 6, title: "某種物質", rating: 7.3, description: "當明星遇上神秘藥物，黑市交易背後的祕密。"),
    Movie(rank: 7, title: "美國內戰", rating: 7.0, description: "史詩級內戰再現，愛恨情仇盡在其中。"),
    Movie(rank: 8, title: "怪奇大法師", rating: 7.8, description: "魔法與奇幻的世界裡，大法師帶你穿越時空。"),
    Movie(rank: 9, title: "猩球崛起：新世界", rating: 7.4, description: "猩猩的世界再度來襲，人類與猩猩的命運交錯。"),
    Movie(rank: 10, title: "長腿", rating: 7.4, description: "一段浪漫又充滿成長挑戰的故事，讓人哭笑不得。"),
  ];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("2024 電影排行榜"),
      ),
      body: ListView.builder(
        itemCount: movies.length,
        itemBuilder: (context, index) {
          final movie = movies[index];
          return Padding(
            padding: const EdgeInsets.symmetric(vertical: 8.0, horizontal: 16.0),
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                // 顯示排名與電影名稱
                Row(
                  children: [
                    Text(
                      "#${movie.rank}",
                      style: TextStyle(fontSize: 14, color: Colors.grey),
                    ),
                    SizedBox(width: 8),
                    Text(
                      movie.title,
                      style: TextStyle(fontSize: 20, fontWeight: FontWeight.bold),
                    ),
                  ],
                ),
                SizedBox(height: 4),
                // 顯示評分
                Text(
                  "評分：${movie.rating.toStringAsFixed(1)}",
                  style: TextStyle(fontSize: 14, color: Colors.blue),
                ),
                SizedBox(height: 4),
                // 顯示電影描述
                Text(
                  movie.description,
                  style: TextStyle(fontSize: 16, color: Colors.grey[600]),
                ),
                Divider(), // 加個分隔線，讓每個項目都井然有序
              ],
            ),
          );
        },
      ),
    );
  }
}

void main() {
  runApp(
    MaterialApp(
      debugShowCheckedModeBanner: false,
      home: MovieListPage(),
    ),
  );
}
