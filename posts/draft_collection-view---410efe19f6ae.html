collection view實作

  1. 在storyboard建立UICollectionViewController物件
  2. 定義自訂的DemoCollectionViewController(繼承UICollectionViewController)，將storyboard的UICollectionViewController物件類別設為DemoCollectionViewController
  3. 定義排版和cell大小間距 
override func viewDidLoad() {
super.viewDidLoad()
let layout: UICollectionViewFlowLayout = self.collectionViewLayout as UICollectionViewFlowLayout
layout.minimumInteritemSpacing = 2
layout.minimumLineSpacing = 2

switch UIScreen.mainScreen().bounds.size.width {
case 320:
layout.itemSize = CGSizeMake(105, 105)
case 375:
layout.itemSize = CGSizeMake(123, 123)
case 414:
layout.itemSize = CGSizeMake(136, 136)
default:
break
}
}


4. 定義section和cell個數

override func numberOfSectionsInCollectionView(collectionView: UICollectionView) -> Int {
return 1
}
override func collectionView(collectionView: UICollectionView, numberOfItemsInSection section: Int) -> Int {
return 20
}

5. 定義cell內容

override func collectionView(collectionView: UICollectionView, cellForItemAtIndexPath indexPath: NSIndexPath) -> UICollectionViewCell {
let cell = collectionView.dequeueReusableCellWithReuseIdentifier(reuseIdentifier, forIndexPath: indexPath) as DemoCollectionViewCell

// Configure the cell
cell.imageView.image = UIImage(named: “people”)
return cell
}
