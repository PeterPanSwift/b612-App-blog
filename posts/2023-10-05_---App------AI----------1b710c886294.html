UIKit iOS App

class SettingsViewController: UITableViewController {
    
    let sections = [
        [["icon": "person.circle", "title": "Sign in to your iPhone", "subtitle": "Set up iCloud, the App Store, and more."]],
        [["icon": "globe", "title": "VPN"], ["icon": "hourglass", "title": "Screen Time"], ["icon": "play.rectangle", "title": "Action Button"]],
        [["icon": "gear", "title": "General"], ["icon": "person.crop.circle", "title": "Accessibility"], ["icon": "hand.raised", "title": "Privacy & Security"]],
        [["icon": "lock", "title": "Passwords"], ["icon": "safari", "title": "Safari"], ["icon": "newspaper", "title": "News"], ["icon": "arrow.right.arrow.left.circle", "title": "Translate"], ["icon": "map", "title": "Maps"]]
    ]
    
    override func viewDidLoad() {
        super.viewDidLoad()
        
        title = "Settings"
        tableView.register(UITableViewCell.self, forCellReuseIdentifier: "cell")
    }

    override func numberOfSections(in tableView: UITableView) -> Int {
        return sections.count
    }

    override func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        return sections[section].count
    }
    
    override func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "cell", for: indexPath)
        let item = sections[indexPath.section][indexPath.row]
        
        cell.textLabel?.text = item["title"]
        cell.imageView?.image = UIImage(systemName: item["icon"] ?? "")
        
        if let subtitle = item["subtitle"] {
            cell.detailTextLabel?.text = subtitle
            cell.detailTextLabel?.textColor = .gray
        }
        
        return cell
    }
}


Flutter App

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      theme: ThemeData(
        primarySwatch: Colors.blue,
        visualDensity: VisualDensity.adaptivePlatformDensity,
      ),
      home: SettingsScreen(),
    );
  }
}

class SettingsScreen extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Settings'),
      ),
      body: ListView(
        children: [
          _buildProfileSection(),
          _buildSettingItem(Icons.vpn_lock, 'VPN'),
          _buildSettingItem(Icons.timer, 'Screen Time'),
          _buildSettingItem(Icons.play_arrow, 'Action Button'),
          Divider(),
          _buildSettingItem(Icons.settings, 'General'),
          _buildSettingItem(Icons.accessibility_new, 'Accessibility'),
          _buildSettingItem(Icons.security, 'Privacy & Security'),
          Divider(),
          _buildSettingItem(Icons.vpn_key, 'Passwords'),
          _buildSettingItem(Icons.language, 'Safari'),
          _buildSettingItem(Icons.library_books, 'News'),
          _buildSettingItem(Icons.translate, 'Translate'),
          _buildSettingItem(Icons.map, 'Maps'),
        ],
      ),
    );
  }

  Widget _buildProfileSection() {
    return ListTile(
      leading: CircleAvatar(
        radius: 25,
        backgroundColor: Colors.grey[300],
      ),
      title: Text('Sign in to your iPhone'),
      subtitle: Text('Set up iCloud, the App Store, and more.'),
    );
  }

  Widget _buildSettingItem(IconData icon, String title) {
    return ListTile(
      leading: Icon(icon, color: Colors.blue),
      title: Text(title),
      trailing: Icon(Icons.chevron_right),
    );
  }
}
