struct NumberKey: EnvironmentKey {
    static let defaultValue = 0
}

extension EnvironmentValues {
    var number: Int {
        get {
            self[NumberKey.self]
        }
        set {
            self[NumberKey.self] = newValue
        }
    }
}

extension View {
    func number(_ number: Int) -> some View {
        environment(\.number, number)
    }
}
