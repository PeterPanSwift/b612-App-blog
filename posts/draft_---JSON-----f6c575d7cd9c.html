import Foundation
struct SearchResponse: Codable {
    let resultCount: Int
    let results: [StoreItem]
}
struct StoreItem: Codable {
    let artistName: String
    let trackName: String
    let collectionName: String?
    let previewUrl: URL
    let artworkUrl100: URL
    let trackPrice: Double?
    let releaseDate: Date
    let isStreamable: Bool?
}
protocol URLSessionDecodable {
    init(from output: Data) throws
}
extension Array: URLSessionDecodable where Element == StoreItem {
    init(from output: Data) throws {
        let decoder = JSONDecoder()
        decoder.dateDecodingStrategy = .iso8601
        let searchResponse = try decoder.decode(SearchResponse.self, from: output)
        self.init(searchResponse.results)
    }
}
let urlStr = "https://itunes.apple.com/search?term=swift&media=music"
if let url = URL(string: urlStr) {
    URLSession.shared.dataTask(with: url) { data, response , error in
        if let data = data {
            do {
                let storeItems = try [StoreItem](from: data)
                print(storeItems)
            } catch {
                print(error)
            }
        } else {
            print("error")
        }
    }.resume()
}
