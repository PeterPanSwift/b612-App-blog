  • 請用 SwiftUI 寫一個猜數字 App。
import SwiftUI

struct GuessingGameView: View {
    @State private var randomNumber = Int.random(in: 1...100)
    @State private var guess = ""
    @State private var attempts = 0
    @State private var resultMessage = ""

    var body: some View {
        VStack {
            Text("Guess the number between 1 and 100:")
                .font(.title)
                .padding()

            TextField("Enter your guess", text: $guess)
                .keyboardType(.numberPad)
                .padding()

            Button("Guess") {
                guard let guessNumber = Int(guess) else {
                    resultMessage = "Please enter a valid number."
                    return
                }

                attempts += 1

                if guessNumber == randomNumber {
                    resultMessage = "Congratulations! You guessed the number in \(attempts) attempts."
                } else if guessNumber < randomNumber {
                    resultMessage = "Higher!"
                } else {
                    resultMessage = "Lower!"
                }

                guess = ""
            }
            .padding()

            Text(resultMessage)
                .padding()

            Button("Reset") {
                resetGame()
            }
            .padding()
        }
        .padding()
    }

    func resetGame() {
        randomNumber = Int.random(in: 1...100)
        guess = ""
        attempts = 0
        resultMessage = ""
    }
}

@main
struct GuessingGameApp: App {
    var body: some Scene {
        WindowGroup {
            GuessingGameView()
        }
    }
}
