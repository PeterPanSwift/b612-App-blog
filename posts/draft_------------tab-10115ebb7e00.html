動態顯示 / 隱藏某個 tab 


import UIKit
class ViewController: UIViewController {
    
    var tabViewControllers: [UIViewController]?
override func viewDidLoad() {
        super.viewDidLoad()
        // Do any additional setup after loading the view.
    }
@IBAction func toggleSearchTab(_ sender: Any) {
        if tabViewControllers == nil {
            tabViewControllers = tabBarController?.viewControllers
        }
        if tabBarController?.viewControllers?.count == tabViewControllers?.count {
            tabBarController?.viewControllers?.remove(at: 1)
        } else {
            tabBarController?.viewControllers = tabViewControllers
        }
    }
    
}
