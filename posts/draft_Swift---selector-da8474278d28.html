Swift裡使用selector 

override func viewDidLoad() {
 super.viewDidLoad() 
 var but = UIButton(frame: CGRectMake(10, 10, 100, 30))
 but.setTitle(“test”, forState: .Normal)
 but.setTitleColor(UIColor.blackColor(), forState: .Normal)
view.addSubview(but)
but.addTarget(self, action: “test:arg2:”, forControlEvents: UIControlEvents.TouchUpInside)
}
func test(arg1:AnyObject, arg2:AnyObject)
{
 println(“test”)
}
