    case invalidURL
    case invalidStatusCode

定義串接 API 抓資料的 fetchItems

利用 https://app.requestly.io/delay/2000 增加 delay。

func fetchItems(term: String) async throws -> [StoreItem] {
        let logger = Logger()
        logger.log("fetch \(term)")
        guard let urlString = "https://app.requestly.io/delay/2000/https://itunes.apple.com/search?term=\(term)&media=music&country=tw".addingPercentEncoding(withAllowedCharacters: .urlQueryAllowed),
              let url = URL(string: urlString) else {
            throw NetworkError.invalidURL
        }
        let (data, response) = try await URLSession.shared.data(from: url)
        guard let httpResponse = response as? HTTPURLResponse,
              httpResponse.statusCode == 200 else {
            throw NetworkError.invalidStatusCode
        }
        let searchResponse = try JSONDecoder().decode(SearchResponse.self, from: data)
        logger.log("\(term) \(searchResponse.results.randomElement()?.trackName ?? "")")
        return searchResponse.results
}

        do {
            items1 = try await fetchItems(term: "劉德華")

同時抓取多個 API 的資料

  • 實驗 1
var items1 = [StoreItem]()
var items2 = [StoreItem]()
    
func fetchData() async {
        async let items1 = fetchItems(term: "劉德華")
        async let items2 = fetchItems(term: "梁朝偉")
        do {
            (self.items1, self.items2) = try await (items1, items2)
        } catch {
            
        }
}
