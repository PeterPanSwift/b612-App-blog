控制 image view 圖片顯示區塊的 contentsRect


let imageView = UIImageView(frame: CGRect(x: 100, y: 100, width: 200, height: 100))
        imageView.image = UIImage(named: "peter")
        view.addSubview(imageView)
        
        let imageView2 = UIImageView(frame: CGRect(x: 100, y: 250, width: 200, height: 100))
        imageView2.image = UIImage(named: "peter")
        imageView2.layer.contentsRect = CGRect(x: 0, y: 0, width: 0.5, height: 0.5)
        view.addSubview(imageView2)
        
        let imageView3 = UIImageView(frame: CGRect(x: 100, y: 400, width: 200, height: 100))
        imageView3.image = UIImage(named: "peter")
        imageView3.layer.contentsRect = CGRect(x: 0, y: 0, width: 0.5, height: 0.5)
        imageView3.layer.contentsGravity = .resizeAspect
        view.addSubview(imageView3)
        
        
        let imageView4 = UIImageView(frame: CGRect(x: 100, y: 550, width: 200, height: 100))
        imageView4.image = UIImage(named: "peter")
        imageView4.layer.contentsRect = CGRect(x: 0, y: 0, width: 0.5, height: 0.5)
        imageView4.layer.contentsGravity = .resizeAspectFill
        imageView4.layer.masksToBounds = true
        view.addSubview(imageView4)

let imageView = UIImageView(frame: CGRect(x: 100, y: 100, width: 200, height: 200))
        let imageViewRatio = imageView.bounds.height / imageView.bounds.width
        imageView.image = UIImage(named: "peter")
        view.addSubview(imageView)
        if let image = imageView.image {
            let widthRatio = 0.5
            let heightRatio = image.size.width * widthRatio * imageViewRatio / image.size.height
            imageView.layer.contentsRect = CGRect(x: 0, y: 0, width: widthRatio, height: heightRatio)
}
