請 AI 寫一個回傳隨機名字的 API

from flask import Flask
import random

app = Flask(__name__)

@app.route('/random_name', methods=['GET'])
def get_random_name():
    names = ['John', 'Jane', 'Tom', 'Sue', 'Steve', 'Anna']
    return random.choice(names)

if __name__ == "__main__":
    app.run(debug=True)
