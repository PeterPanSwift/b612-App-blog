In terms of improvements, the iPhone 13 has a faster A15 Bionic chip, improved camera system, longer battery life, and a smaller notch compared to the iPhone 12.

As for Swift inheritance, let’s say we have a class called Phone with properties such as model and color. We can create a subclass called iPhone12 that inherits from Phone and adds its own properties such as faceID and screenSize. Similarly, we can create another subclass called iPhone13 that inherits from iPhone12 and adds its own properties such as proMotionDisplay and cinematicMode.

Here’s an example code snippet to illustrate this:

class Phone {
    var model: String
    var color: String
    
    init(model: String, color: String) {
        self.model = model
        self.color = color
    }
}

class iPhone12: Phone {
    var faceID: Bool
    var screenSize: Double
    
    init(model: String, color: String, faceID: Bool, screenSize: Double) {
        self.faceID = faceID
        self.screenSize = screenSize
        super.init(model: model, color: color)
    }
}

class iPhone13: iPhone12 {
    var proMotionDisplay: Bool
    var cinematicMode: Bool
    
    init(model: String, color: String, faceID: Bool, screenSize: Double, proMotionDisplay: Bool, cinematicMode: Bool) {
        self.proMotionDisplay = proMotionDisplay
        self.cinematicMode = cinematicMode
        super.init(model: model, color: color, faceID: faceID, screenSize: screenSize)
    }
}
