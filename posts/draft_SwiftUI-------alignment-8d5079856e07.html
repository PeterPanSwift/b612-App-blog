SwiftUI 調整圖片的 alignment


struct ContentView: View {
    var body: some View {
        Image("m3Max")
            .resizable()
            .scaledToFill()
            .frame(width: 300, height: 100, alignment: .customBottom)


            .clipped()
    }
}

extension VerticalAlignment {
    private struct CustomBottomAlignment: AlignmentID {
        static func defaultValue(in context: ViewDimensions) -> CGFloat {
            return context.height * 0.2 // 75% from the top
        }
    }
    
    static let customBottom = VerticalAlignment(CustomBottomAlignment.self)
}

extension Alignment {
    static let customBottom = Alignment(horizontal: .center, vertical: .customBottom)
}
