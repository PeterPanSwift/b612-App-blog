var timeObserverToken: Any?
    
    func removePeriodicTimeObserver() {
        if let timeObserverToken = timeObserverToken {
            player.removeTimeObserver(timeObserverToken)
            self.timeObserverToken = nil
        }
    }
    
    func observeCurrentTime() {
        timeObserverToken = player.addPeriodicTimeObserver(forInterval: CMTimeMake(value: 1, timescale: 1), queue: DispatchQueue.main, using: { (CMTime) in
            if self.player.currentItem?.status == .readyToPlay {
                let currentTime = CMTimeGetSeconds(self.player.currentTime())
                self.musicSlider.value = Float(currentTime)
                self.currentTime.text = self.formatConversion(time: currentTime)
                
                //  self.remainTime.text = "-\(self.formatTime(time: currentTime))"
                
                self.nowPlayingInfo[MPNowPlayingInfoPropertyElapsedPlaybackTime] = currentTime
                MPNowPlayingInfoCenter.default().nowPlayingInfo = self.nowPlayingInfo
            }
        })
    }
func formatConversion(time:Float64) -> String {
        let songLength = Int(time)
        let minutes = Int(songLength / 60) // 求 songLength 的商，為分鐘數
        let seconds = Int(songLength % 60) // 求 songLength 的餘數，為秒數
        var time = ""
        if minutes < 10 {
            time = "0\(minutes):"
        } else {
            time = "\(minutes)"
        }
        if seconds < 10 {
            time += "0\(seconds)"
        } else {
            time += "\(seconds)"
        }
        return time
    }



@IBAction func setTime(_ sender: UISlider, forEvent event: UIEvent) {
        let touch = event.allTouches?.first
        switch touch?.phase {
        case .began:
            self.removePeriodicTimeObserver()
        case .moved:
            let seconds = Int64(sender.value)
            let targetTime:CMTime = CMTimeMake(value: seconds, timescale: 1)
            player.seek(to: targetTime)
        case .ended:
            self.observeCurrentTime()
        default:
            break
        }
        
    }
