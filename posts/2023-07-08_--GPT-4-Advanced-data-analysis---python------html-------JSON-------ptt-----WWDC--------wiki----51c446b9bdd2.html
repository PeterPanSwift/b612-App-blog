import json

# Function to extract information from a row
def extract_info(row):
    # Initialize a dictionary to store the information
    info = {}

    # Extract the title and link
    title_div = row.find('div', class_='title')
    title_link = title_div.find('a')
    if title_link is not None:
        info['title'] = title_link.text.strip()
        info['link'] = "https://www.ptt.cc" + title_link.get('href')
    else:
        info['title'] = title_div.text.strip()
        info['link'] = None

    # Extract the author
    author_div = row.find('div', class_='author')
    info['author'] = author_div.text.strip()

    # Extract the date
    date_div = row.find('div', class_='date')
    info['date'] = date_div.text.strip()

    # Extract the number of recommendations
    nrec_div = row.find('div', class_='nrec')
    nrec_span = nrec_div.find('span')
    if nrec_span is not None:
        info['nrec'] = nrec_span.text.strip()
    else:
        info['nrec'] = ''

    return info

# Loop through all the rows and extract information
articles = [extract_info(row) for row in rows]

# Convert the list of dictionaries to a JSON string
json_articles = json.dumps(articles, ensure_ascii=False, indent=4)

# Loop through each video section and extract information
for video in video_sections:
    # Extract video title
    title_section = video.find('h4', {'class': 'no-margin-bottom video-title'})
    title = title_section.text.strip() if title_section else None

    # Extract video description
    description_section = video.find('p', {'class': 'description'})
    description = description_section.text.strip() if description_section else None

    # Extract video URL
    video_url_section = video.find('a', {'class': 'video-image-link'})
    video_url = video_url_section['href'] if video_url_section else None

    # Extract image URL
    image_url_section = video.find('img', {'class': 'video-image medium-scale'})
    image_url = image_url_section['src'] if image_url_section else None

    # Extract video duration
    duration_section = video.find('span', {'class': 'video-duration'})
    duration = duration_section.text.strip() if duration_section else None

    # Extract WWDC year
    wwdc_year_section = video.find('span', {'class': 'smaller'})
    wwdc_year = wwdc_year_section.text.strip() if wwdc_year_section else None

    # Append the video data to the list
    video_data.append({
        'title': title,
        'description': description,
        'video_url': video_url,
        'image_url': image_url,
        'duration': duration,
        'wwdc_year': wwdc_year,
    })

# Print the first few items to verify extraction
