import EventKit

let isFullAccessAuthorized: Bool
if #available(iOS 17.0, *) {
    isFullAccessAuthorized = EKEventStore.authorizationStatus(for: .event) == .fullAccess
} else {
    // Fall back on earlier versions.
    isFullAccessAuthorized = EKEventStore.authorizationStatus(for: .event) == .authorized
}
let eventStore = EKEventStore()
let start = Date.now
let end = Calendar.current.date(byAdding: .month, value: 1, to: Date.now) ?? Date()
let predicate = eventStore.predicateForEvents(withStart: start, end: end, calendars: nil)
let events = eventStore.events(matching: predicate).sorted(by: { (first: EKEvent, second: EKEvent) in
    return first.compareStartDate(with: second) == .orderedAscending
})
