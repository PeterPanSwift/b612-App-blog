func scene(_ scene: UIScene, openURLContexts URLContexts: Set<UIOpenURLContext>) {
        if let url = URLContexts.first?.url {
            let components = URLComponents(url: url, resolvingAgainstBaseURL: true)
            
            print(components?.queryItems?.first?.value)
            if let code = components?.queryItems?.first?.value {
                let url = URL(string: "https://dribbble.com/oauth/token")!
                var request = URLRequest(url: url)
                request.httpMethod = "POST"
                request.setValue("application/json", forHTTPHeaderField: "Content-Type")
                let tokenParameters = TokenParameters(clientId: "e24008484c4d0f66b3a08dbbf08b86422d7b4d2f4a17b2f93a360743809fcc3b", clientSecret: "58192f1c8fb55ac6032fab9ec8f103245119b60f4d23604a8b3676a13dc3d98f", code: code)
                let encoder = JSONEncoder()
                encoder.keyEncodingStrategy = .convertToSnakeCase
                if let data = try? encoder.encode(tokenParameters) {
                    request.httpBody = data
                    print(String(data: data, encoding: .utf8))
                }
                URLSession.shared.dataTask(with: request) { (data, response, error) in
                    let decoder = JSONDecoder()
                    decoder.keyDecodingStrategy = .convertFromSnakeCase
                    print(String(data: data!, encoding: .utf8))
                    if let data = data,
                       let tokenResponse = try? decoder.decode(TokenResponse.self, from: data){
                        print("accessToken", tokenResponse.accessToken)
                    }
                }.resume()
            }
        }
       
    }
