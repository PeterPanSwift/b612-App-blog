struct Song: Identifiable {
    let id = UUID()
    let name: String
    let singer: String
}
struct ContentView: View {
    @State private var songs = [
        Song(name: "真我的風采", singer: "劉德華"),
        Song(name: "回到你身邊", singer: "劉德華"),
        Song(name: "相思成災", singer: "劉德華"),
        Song(name: "在乎您", singer: "劉德華"),
        Song(name: "真情難收", singer: "劉德華")
    ]
    var body: some View {
        NavigationView {
            List {
                ForEach(songs) { song in
                    ZStack {
                        SongRow(song: song)
                        
                        NavigationLink {
                            DetailView(song: song)
                        } label: {
                            EmptyView()
                        }
                        .opacity(0)
                    }
                   
                }
            }
        }
    }
}
struct SongRow: View {
    let song: Song
    var body: some View {
        HStack {
            Image(song.name)
                .resizable()
                .scaledToFill()
                .frame(width: 100, height: 100)
                .clipShape(Circle())
            Spacer()
            Text("\(song.name)")
            
        }
    }
}
