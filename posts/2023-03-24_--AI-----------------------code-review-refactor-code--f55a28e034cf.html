class ViewController: UIViewController {
    var scoreA : Int = 0
    var scoreB : Int = 0
    var scoreA2 : Int = 0
    var scoreB2 : Int = 0
    var player : String = ""
    
    @IBOutlet weak var playerName1: UILabel!
    
    @IBOutlet weak var playerName2: UILabel!
    
    @IBOutlet weak var scoreLabelA: UILabel!
    
    @IBOutlet weak var scoreLabelB: UILabel!
 
    
    
    override func viewDidLoad() {
        super.viewDidLoad()

        NewGame()
        
    }


    @IBAction func getPointButtonA(_ sender: UIButton) {
  
        switch sender.tag{
        case 1 :
            scoreA += 1
            scoreA2 = scoreA
            getPoint()
        case 2 :
            scoreB += 1
            scoreB2 = scoreB
            getPoint()
        default : return
        }
    }
    
    
    @IBAction func retrunButton(_ sender: UIButton) {
        
        switch sender.tag{
        case 1 :
            scoreA -= 1
            scoreA2 = scoreA
            getPoint()
        case 2 :
            scoreB -= 1
            scoreB2 = scoreB
            getPoint()
        default : return
        }
        
    }
    

    
    @IBAction func NewGameButton(_ sender: Any) {
        NewGame()
    }
    
    @IBAction func changeButton(_ sender: Any) {
 
        scoreA = scoreB
        scoreB = scoreA2
        scoreA2 = scoreB2
        scoreB2 = scoreB
        scoreLabelA.text = String(scoreA2)
        scoreLabelB.text = String(scoreB2)
        if playerName1.text == "player 1"{
            playerName1.text = "player 2"
            playerName2.text = "player 1"
        }else{
            playerName1.text = "player 1"
            playerName2.text = "player 2"
        }
        
    }
    
    
    func NewGame(){
        scoreA = 0
        scoreB = 0
        scoreA2 = 0
        scoreB2 = 0
        scoreLabelA.text = "0"
        scoreLabelB.text = "0"
    }
    
    func gameEndAlert(){
        if scoreA2 > scoreB2 {
           player = "player 1"
        }else{ player = "player 2" }
        let endAlert = UIAlertController(title: "Game End", message: "Winner is \(player)", preferredStyle: .alert)
        let endAcrion = UIAlertAction(title: "Start New Game", style: .default, handler: {
            (action: UIAlertAction!) -> Void in
            self.NewGame()
        })
        
        endAlert.addAction(endAcrion)
        
        present(endAlert, animated: true)
    }
    
    func getPoint(){
        
        scoreLabelA.text = String(scoreA2)
        scoreLabelB.text = String(scoreB2)
        
        let pointDiff = abs(scoreA2-scoreB2)
        
        if pointDiff > 1{
            if scoreA2 > 10 || scoreB2 > 10{
                gameEndAlert()
            }
        }
    }
    
    
}

以下是學生可以改進的地方：

  1. 變數命名：
    使用具有更明確意義的變數名稱，例如將 scoreA2 和 scoreB2 改為 displayScoreA 和 displayScoreB。
  2. 代碼重複：
    在 getPointButtonA 和 retrunButton 方法中，有重複的代碼。可以考慮將共同的代碼部分提取成一個單獨的方法，例如 updateScore，並將相應的參數（如分數增減和玩家）傳遞給這個方法。這樣可以讓代碼更簡潔、易於維護。
  3. 方法命名：
    使用更具描述性的方法名稱。例如，將 getPoint 改為 updateScoreLabels，將 changeButton 改為 swapPlayersAndScores。
  4. 常量和魔法數字：
    在 getPoint 方法中，應使用常量代替硬編碼的數字（如10）。這樣，如果需要更改閾值，只需更改一個地方即可。例如，可以在類的開始處定義一個常量 let winningScore = 10，然後在 getPoint 方法中使用它。
  5. 界面元素命名：
    將界面元素的名稱改為更具描述性的名稱。例如，將 playerName1 改為 player1NameLabel，將 scoreLabelA 改為 player1ScoreLabel。
  6. 代碼格式和風格：
    確保整個程式的縮排和空格保持一致。這有助於提高代碼的可讀性和易於維護。
    將相關的界面元素和方法分組，以保持代碼結構的清晰。例如，將所有 IBOutlet 和 IBAction 放在一起，並將功能相似的方法放在一起。
  7. 適當使用 switch 語句：
    在 getPointButtonA 和 retrunButton 方法中，可以使用 if-else 語句代替 switch 語句，因為這裡只有兩個情況。
if sender.tag == 1 {
    // 代碼
} else if sender.tag == 2 {
    // 代碼
}

在設定 player 變量時，可以使用三元運算符簡化代碼：

player = scoreA2 > scoreB2 ? "player 1" : "player 2"
