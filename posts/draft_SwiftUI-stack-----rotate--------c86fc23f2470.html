    var body: some View {
        HStack(spacing: 0) {
            Text("3")
                .padding(EdgeInsets(top: 5, leading: 20, bottom: 5, trailing: 20))
                .background(content: {
                    Color.red
                })
            Text("6")
                .padding(EdgeInsets(top: 5, leading: 20, bottom: 5, trailing: 20))
                .background {
                    Color.black
                }
        }
        .foregroundColor(.white)
    }
}
